29inline vec3 toHSL(const Color color) {
30 const auto cmax = comp_max(color);
31 const auto cmin = comp_min(color);
32 const auto delta = cmax - cmin;
33
34 vec3 hsl = {0.f, 0.f, (cmax + cmin) / 2.f};
35
36 if (abs(delta) <= EPSILON) {
37 return hsl;
38 }
39
40 if (abs(cmax - color.r) <= EPSILON) {
41 hsl.x = static_cast<float>(std::fmod((color.g - color.b) / delta, 6.f));
42 } else if (abs(cmax - color.g) <= EPSILON) {
43 hsl.x = (color.b - color.r) / delta + 2.f;
44 } else {
45 hsl.x = (color.r - color.g) / delta + 4.f;
46 }
47
48 hsl.y = delta / (1.f - std::abs(2.f * hsl.z - 1.f));
49 hsl.x /= 6.f;
50 hsl.x = std::fmod(hsl.x + 1.f, 1.f);
51 return hsl;
52}

54inline Color toRGB(const vec3& hsl) {
55 const float k = hsl.x * 6.f;
56 const float c = (1.f - std::abs(2.f * hsl.z - 1.f)) * hsl.y;
57 const float x = c * (1.f - std::abs(std::fmod(k, 2.f) - 1.f));
58 const float m = hsl.z - c / 2.f;
59 const int d = int(std::floor(k));
60
61 vec3 rgb;
62
63 switch (d) {
64 case 0:
65 rgb = {c, x, 0.f};
66 break;
67 case 1:
68 rgb = {x, c, 0.f};
69 break;
70 case 2:
71 rgb = {0.f, c, x};
72 break;
73 case 3:
74 rgb = {0.f, x, c};
75 break;
76 case 4:
77 rgb = {x, 0.f, c};
78 break;
79 default:
80 rgb = {c, 0.f, x};
81 break;
82 }
83
84 return Color{
85 (unsigned char) (255 * (rgb.x + m)),
86 (unsigned char) (255 * (rgb.y + m)),
87 (unsigned char) (255 * (rgb.z + m)),
88 (unsigned char) 255,
89 };
90}
