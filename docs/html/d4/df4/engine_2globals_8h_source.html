8template<typename Str>
9[[nodiscard]] constexpr size_t hashString(const Str& toHash) {
10 // NOTE: For this example, I'm requiring size_t to be 64-bit, but you could
11 // easily change the offset and prime used to the appropriate ones
12 // based on sizeof(size_t).
13 static_assert(sizeof(size_t) == 8);
14 // FNV-1a 64 bit algorithm
15 size_t result = 0xcbf29ce484222325; // FNV offset basis
16
17 for (char c : toHash) {
18 result ^= c;
19 result *= 1099511628211; // FNV prime
20 }
21 return result;
22}
