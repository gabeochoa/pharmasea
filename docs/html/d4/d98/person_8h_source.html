152 std::tuple<int, int> get_face_direction(vec3 new_pos_x, vec3 new_pos_z) {
153 int facedir_x = -1;
154 int facedir_z = -1;
155
156 vec3 delta_distance_x = new_pos_x - this->get<Transform>().raw_position;
157 if (delta_distance_x.x > 0) {
158 facedir_x = Transform::FrontFaceDirection::RIGHT;
159 } else if (delta_distance_x.x < 0) {
160 facedir_x = Transform::FrontFaceDirection::LEFT;
161 }
162
163 vec3 delta_distance_z = new_pos_z - this->get<Transform>().raw_position;
164 if (delta_distance_z.z > 0) {
165 facedir_z = Transform::FrontFaceDirection::FORWARD;
166 } else if (delta_distance_z.z < 0) {
167 facedir_z = Transform::FrontFaceDirection::BACK;
168 }
169 return std::make_tuple(facedir_x, facedir_z);
170 }
