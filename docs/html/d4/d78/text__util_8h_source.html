23static void DrawTextCodepoint3D(Font font, int codepoint, vec3 position,
24 float fontSize, bool backface, Color tint) {
25 // Character index position in sprite font
26 // NOTE: In case a codepoint is not available in the font, index returned
27 // points to '?'
28 int index = GetGlyphIndex(font, codepoint);
29 float scale = fontSize / (float) font.baseSize;
30
31 // Character destination rectangle on screen
32 // NOTE: We consider charsPadding on drawing
33 position.x += (float) (font.glyphs[index].offsetX - font.glyphPadding) /
34 (float) font.baseSize * scale;
35 position.z += (float) (font.glyphs[index].offsetY - font.glyphPadding) /
36 (float) font.baseSize * scale;
37
38 // Character source rectangle from font texture atlas
39 // NOTE: We consider chars padding when drawing, it could be required for
40 // outline/glow shader effects
41 Rectangle srcRec = {font.recs[index].x - (float) font.glyphPadding,
42 font.recs[index].y - (float) font.glyphPadding,
43 font.recs[index].width + 2.0f * font.glyphPadding,
44 font.recs[index].height + 2.0f * font.glyphPadding};
45
46 float width = (float) (font.recs[index].width + 2.0f * font.glyphPadding) /
47 (float) font.baseSize * scale;
48 float height =
49 (float) (font.recs[index].height + 2.0f * font.glyphPadding) /
50 (float) font.baseSize * scale;
51
52 if (font.texture.id > 0) {
53 const float x = 0.0f;
54 const float y = 0.0f;
55 const float z = 0.0f;
56
57 // normalized texture coordinates of the glyph inside the font texture
58 // (0.0f -> 1.0f)
59 const float tx = srcRec.x / font.texture.width;
60 const float ty = srcRec.y / font.texture.height;
61 const float tw = (srcRec.x + srcRec.width) / font.texture.width;
62 const float th = (srcRec.y + srcRec.height) / font.texture.height;
63
64 if (SHOW_LETTER_BOUNDRY)
65 DrawCubeWiresV((vec3){position.x + width / 2, position.y,
66 position.z + height / 2},
67 (vec3){width, LETTER_BOUNDRY_SIZE, height},
68 LETTER_BOUNDRY_COLOR);
69
70 rlCheckRenderBatchLimit(4 + 4 * backface);
71 rlSetTexture(font.texture.id);
72
73 rlPushMatrix();
74 rlTranslatef(position.x, position.y, position.z);
75
76 rlBegin(RL_QUADS);
77 rlColor4ub(tint.r, tint.g, tint.b, tint.a);
78
79 // Front Face
80 rlNormal3f(0.0f, 1.0f, 0.0f); // Normal Pointing Up
81 rlTexCoord2f(tx, ty);
82 rlVertex3f(x, y, z); // Top Left Of The Texture and Quad
83 rlTexCoord2f(tx, th);
84 rlVertex3f(x, y, z + height); // Bottom Left Of The Texture and Quad
85 rlTexCoord2f(tw, th);
86 rlVertex3f(x + width, y,
87 z + height); // Bottom Right Of The Texture and Quad
88 rlTexCoord2f(tw, ty);
89 rlVertex3f(x + width, y, z); // Top Right Of The Texture and Quad
90
91 if (backface) {
92 // Back Face
93 rlNormal3f(0.0f, -1.0f, 0.0f); // Normal Pointing Down
94 rlTexCoord2f(tx, ty);
95 rlVertex3f(x, y, z); // Top Right Of The Texture and Quad
96 rlTexCoord2f(tw, ty);
97 rlVertex3f(x + width, y, z); // Top Left Of The Texture and Quad
98 rlTexCoord2f(tw, th);
99 rlVertex3f(x + width, y,
100 z + height); // Bottom Left Of The Texture and Quad
101 rlTexCoord2f(tx, th);
102 rlVertex3f(x, y,
103 z + height); // Bottom Right Of The Texture and Quad
104 }
105 rlEnd();
106 rlPopMatrix();
107
108 rlSetTexture(0);
109 }

112// Draw a 2D text in 3D space
113static void DrawText3D(Font font, const char* text, vec3 position,
114 float fontSize, float fontSpacing, float lineSpacing,
115 bool backface, Color tint) {
116 unsigned int length =
117 TextLength(text); // Total length in bytes of the text,
118 // scanned by codepoints in loop
119
120 float textOffsetY = 0.0f; // Offset between lines (on line break '\n')
121 float textOffsetX = 0.0f; // Offset X to next character to draw
122
123 float scale = fontSize / (float) font.baseSize;
124
125 for (unsigned int i = 0; i < length;) {
126 // Get next codepoint from byte string and glyph index in font
127 int codepointByteCount = 0;
128 int codepoint = GetCodepoint(&text[i], &codepointByteCount);
129 int index = GetGlyphIndex(font, codepoint);
130
131 // NOTE: Normally we exit the decoding sequence as soon as a bad byte is
132 // found (and return 0x3f) but we need to draw all of the bad bytes
133 // using the '?' symbol moving one byte
134 if (codepoint == 0x3f) codepointByteCount = 1;
135
136 if (codepoint == '\n') {
137 // NOTE: Fixed line spacing of 1.5 line-height
138 // TODO: Support custom line spacing defined by user
139 textOffsetY += scale + lineSpacing / (float) font.baseSize * scale;
140 textOffsetX = 0.0f;
141 } else {
142 if ((codepoint != ' ') && (codepoint != '\t')) {
143 DrawTextCodepoint3D(font, codepoint,
144 (vec3){position.x + textOffsetX, position.y,
145 position.z + textOffsetY},
146 fontSize, backface, tint);
147 }
148
149 if (font.glyphs[index].advanceX == 0)
150 textOffsetX += (float) (font.recs[index].width + fontSpacing) /
151 (float) font.baseSize * scale;
152 else
153 textOffsetX +=
154 (float) (font.glyphs[index].advanceX + fontSpacing) /
155 (float) font.baseSize * scale;
156 }
157
158 i += codepointByteCount; // Move text bytes counter to next codepoint
159 }
160}

162// Measure a text in 3D. For some reason `MeasureTextEx()` just doesn't seem to
163// work so i had to use this instead.
164static vec3 MeasureText3D(Font font, const char* text, float fontSize,
165 float fontSpacing, float lineSpacing) {
166 int len = TextLength(text);
167 int tempLen = 0; // Used to count longer text line num chars
168 int lenCounter = 0;
169
170 float tempTextWidth = 0.0f; // Used to count longer text line width
171
172 float scale = fontSize / (float) font.baseSize;
173 float textHeight = scale;
174 float textWidth = 0.0f;
175
176 int letter = 0; // Current character
177 int index = 0; // Index position in sprite font
178
179 for (int i = 0; i < len; i++) {
180 lenCounter++;
181
182 int next = 0;
183 letter = GetCodepoint(&text[i], &next);
184 index = GetGlyphIndex(font, letter);
185
186 // NOTE: normally we exit the decoding sequence as soon as a bad byte is
187 // found (and return 0x3f) but we need to draw all of the bad bytes
188 // using the '?' symbol so to not skip any we set next = 1
189 if (letter == 0x3f) next = 1;
190 i += next - 1;
191
192 if (letter != '\n') {
193 if (font.glyphs[index].advanceX != 0)
194 textWidth += (font.glyphs[index].advanceX + fontSpacing) /
195 (float) font.baseSize * scale;
196 else
197 textWidth +=
198 (font.recs[index].width + font.glyphs[index].offsetX) /
199 (float) font.baseSize * scale;
200 } else {
201 if (tempTextWidth < textWidth) tempTextWidth = textWidth;
202 lenCounter = 0;
203 textWidth = 0.0f;
204 textHeight += scale + lineSpacing / (float) font.baseSize * scale;
205 }
206
207 if (tempLen < lenCounter) tempLen = lenCounter;
208 }
209
210 if (tempTextWidth < textWidth) tempTextWidth = textWidth;
211
212 vec3 vec = {0};
213 vec.x = tempTextWidth +
214 (float) ((tempLen - 1) * fontSpacing / (float) font.baseSize *
215 scale); // Adds chars spacing to measure
216 vec.y = 0.25f;
217 vec.z = textHeight;
218
219 return vec;
220}

222// Draw a 2D text in 3D space and wave the parts that start with `~~` and end
223// with `~~`. This is a modified version of the original code by @Nighten found
224// here https://github.com/NightenDushi/Raylib_DrawTextStyle
225static void DrawTextWave3D(Font font, const char* text, vec3 position,
226 float fontSize, float fontSpacing, float lineSpacing,
227 bool backface, WaveTextConfig* config, float time,
228 Color tint) {
229 int length = TextLength(text); // Total length in bytes of the text,
230 // scanned by codepoints in loop
231
232 float textOffsetY = 0.0f; // Offset between lines (on line break '\n')
233 float textOffsetX = 0.0f; // Offset X to next character to draw
234
235 float scale = fontSize / (float) font.baseSize;
236
237 bool wave = false;
238
239 for (int i = 0, k = 0; i < length; ++k) {
240 // Get next codepoint from byte string and glyph index in font
241 int codepointByteCount = 0;
242 int codepoint = GetCodepoint(&text[i], &codepointByteCount);
243 int index = GetGlyphIndex(font, codepoint);
244
245 // NOTE: Normally we exit the decoding sequence as soon as a bad byte is
246 // found (and return 0x3f) but we need to draw all of the bad bytes
247 // using the '?' symbol moving one byte
248 if (codepoint == 0x3f) codepointByteCount = 1;
249
250 if (codepoint == '\n') {
251 // NOTE: Fixed line spacing of 1.5 line-height
252 // TODO: Support custom line spacing defined by user
253 textOffsetY += scale + lineSpacing / (float) font.baseSize * scale;
254 textOffsetX = 0.0f;
255 k = 0;
256 } else if (codepoint == '~') {
257 if (GetCodepoint(&text[i + 1], &codepointByteCount) == '~') {
258 codepointByteCount += 1;
259 wave = !wave;
260 }
261 } else {
262 if ((codepoint != ' ') && (codepoint != '\t')) {
263 vec3 pos = position;
264 if (wave) // Apply the wave effect
265 {
266 pos.x += sinf(time * config->waveSpeed.x -
267 k * config->waveOffset.x) *
268 config->waveRange.x;
269 pos.y += sinf(time * config->waveSpeed.y -
270 k * config->waveOffset.y) *
271 config->waveRange.y;
272 pos.z += sinf(time * config->waveSpeed.z -
273 k * config->waveOffset.z) *
274 config->waveRange.z;
275 }
276
277 DrawTextCodepoint3D(
278 font, codepoint,
279 (vec3){pos.x + textOffsetX, pos.y, pos.z + textOffsetY},
280 fontSize, backface, tint);
281 }
282
283 if (font.glyphs[index].advanceX == 0)
284 textOffsetX += (float) (font.recs[index].width + fontSpacing) /
285 (float) font.baseSize * scale;
286 else
287 textOffsetX +=
288 (float) (font.glyphs[index].advanceX + fontSpacing) /
289 (float) font.baseSize * scale;
290 }
291
292 i += codepointByteCount; // Move text bytes counter to next codepoint
293 }
294}
