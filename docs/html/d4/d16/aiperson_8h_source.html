141 virtual void wandering() {
142 switch (job->state) {
143 case Job::State::Initialize: {
144 announce("starting a new wandering job");
145 job->state = Job::State::HeadingToStart;
146 return;
147 }
148 case Job::State::HeadingToStart: {
149 bool arrived = navigate_to(job->start);
150 // TODO we cannot mutate this->job inside navigate because the
151 // `job->state` below will change the new job and not the old
152 // one this foot-gun might be solvable by passing in the global
153 // job to the job processing function, then it wont change until
154 // the next call
155 if (job->path.size() == 0) {
156 personal_queue.push(job);
157 this->job.reset(new Job({
158 .type = Wandering,
159 }));
160 return;
161 }
162 job->state = arrived ? Job::State::WorkingAtStart
163 : Job::State::HeadingToStart;
164 return;
165 }
166 case Job::State::WorkingAtStart: {
167 job->state = Job::State::HeadingToEnd;
168 return;
169 }
170 case Job::State::HeadingToEnd: {
171 bool arrived = navigate_to(job->end);
172 job->state = arrived ? Job::State::WorkingAtEnd
173 : Job::State::HeadingToEnd;
174 return;
175 }
176 case Job::State::WorkingAtEnd: {
177 job->state = Job::State::Completed;
178 return;
179 }
180 case Job::State::Completed: {
181 job->state = Job::State::Completed;
182 return;
183 }
184 }
