116 template<typename T>
117 static constexpr std::shared_ptr<T> getClosestMatchingItem(
118 vec2 pos, float range,
119 std::function<bool(std::shared_ptr<T>)> filter = {}) {
120 float best_distance = range;
121 std::shared_ptr<T> best_so_far;
122 for (auto& e : get_items()) {
123 auto s = dynamic_pointer_cast<T>(e);
124 if (!s) continue;
125 if (filter && !filter(s)) continue;
126 float d = vec::distance(pos, vec::to2(e->position));
127 if (d > range) continue;
128 if (d < best_distance) {
129 best_so_far = s;
130 best_distance = d;
131 }
132 }
133 return best_so_far;
134 }
