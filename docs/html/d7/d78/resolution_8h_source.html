53 void load_resolution_options() {
54#ifdef __APPLE__
55 // Nothing needed here this one works :)
56#else
57 // TODO either implement these for windows or get them in the dll
58 const auto glfwGetPrimaryMonitor = []() -> GLFWmonitor* {
59 return nullptr;
60 };
61 const auto glfwGetVideoModes = [](GLFWmonitor*, int*) -> GLFWvidmode* {
62 return nullptr;
63 };
64#endif
65 GLFWmonitor* monitor = glfwGetPrimaryMonitor();
66 int count = 0;
67 const GLFWvidmode* modes = glfwGetVideoModes(monitor, &count);
68
69 for (int i = 0; i < count; i++) {
70 GLFWvidmode mode = modes[i];
71 // int width
72 // int height
73 // int redBits int greenBits int blueBits
74 // int refreshRate
75
76 // Just kinda easier to not support every possible resolution
77 if (mode.height < 720 || mode.height > 2160) continue;
78
79 options.push_back(
80 ResolutionInfo{.width = mode.width, .height = mode.height});
81 }
82
83 if (options.empty()) {
84 options.push_back(ResolutionInfo{.width = 1280, .height = 720});
85 options.push_back(ResolutionInfo{.width = 1920, .height = 1080});
86 options.push_back(ResolutionInfo{.width = 3860, .height = 2160});
87 }
88
89 // TODO SPEED this kinda slow but it only happens once
90 options.erase(std::unique(options.begin(), options.end()),
