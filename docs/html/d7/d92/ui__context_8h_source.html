139 [[nodiscard]] bool pressed(const InputName& name) {

142 if (a) {
143 eatKey();
144 return a;
145 }
146
147 GamepadButton butt = KeyMap::get_button(STATE, name);
148 bool b = _pressedButtonWithoutEat(butt);
149 if (b) {
150 eatButton();
151 return b;
152 }
153
154 bool c = KeyMap::get_axis(STATE, name)
155 .map([&](GamepadAxisWithDir axis) -> bool {
156 return axis_info.axis == axis.axis &&
157 ((axis.dir - axis_info.dir) >= EPSILON);
158 })
159 .map_error([&](auto exp) {
160 this->handleBadGamepadAxis(exp, STATE, name);
161 })
162 .value_or(false);
163 if (c) {
164 eatAxis();
165 }

235 it++;
236 }
