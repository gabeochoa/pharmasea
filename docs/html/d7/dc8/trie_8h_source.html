30 [[nodiscard]] std::vector<std::string> dump(Node cur,
31 const std::string& prefix) {
32 std::vector<std::string> results;
33
34 std::vector<std::pair<std::string, Node> > q;
35 q.push_back(std::make_pair(prefix, cur));
36 while (!q.empty()) {
37 auto qtop = q.front();
38 q.erase(q.begin());
39 if (qtop.second.is_end) {
40 results.push_back(qtop.first);
41 }
42 for (auto kv : qtop.second.children) {
43 q.push_back(std::make_pair(qtop.first + kv.first, kv.second));
44 }
45 }
46 return results;
47 }

49 [[nodiscard]] std::vector<std::string> dump(
50 const std::string& prefix = "") {
51 auto cur = subtrie(root, prefix);
52 if (cur.has_value()) return dump(*cur, prefix);
53 return std::vector<std::string>();
54 }

56 private:
57 void add_(Node& cur, const std::string& phrase) {
58 if (phrase.empty()) {
59 cur.is_end = true;
60 return;
61 }
62 char c = phrase[0];
63 if (cur.children.find(c) == cur.children.end())
64 cur.children.insert(std::make_pair(c, Node(c)));
65 add_(cur.children.at(c), phrase.substr(1));
66 }
67
68 void reset() { root = Node(0); }
69
70 std::optional<Node> subtrie(Node cur, const std::string& phrase) {
71 if (phrase.empty()) return Node(cur);
72 char c = phrase[0];
73 if (cur.children.find(c) == cur.children.end())
74 return std::optional<Node>();
75 return subtrie(cur.children.at(c), phrase.substr(1));
76 }
77
78 bool contains_(Node cur, const std::string& phrase) {
79 auto resp = subtrie(cur, phrase);
80 if (resp.has_value()) {
81 return resp->is_end;
82 }
83 return false;
84 }
