121inline void render_normal(std::shared_ptr<Entity> entity, float dt) {
122 // Ghost player cant render during normal mode
123 if (entity->has<CanBeGhostPlayer>() &&
124 entity->get<CanBeGhostPlayer>().is_ghost()) {
125 return;
126 }
127
128 if (entity->has<CanBeHighlighted>() &&
129 entity->get<CanBeHighlighted>().is_highlighted) {
130 bool used = render_model_highlighted(entity, dt);
131 if (!used) {
132 render_simple_highlighted(entity, dt);
133 }
134 return;
135 }
136
137 bool used = render_model_normal(entity, dt);
138 if (!used) {
139 render_simple_normal(entity, dt);
140 }
141}
