176inline void collect_user_input(std::shared_ptr<Entity> entity, float dt) {
177 if (entity->is_missing<CollectsUserInput>()) return;
178 CollectsUserInput& cui = entity->get<CollectsUserInput>();
179
180 // Theres no players not in game menu state,
181 const menu::State state = menu::State::Game;
182
183 float left, right, up, down;
184
185 float key_left = KeyMap::is_event(state, InputName::PlayerLeft);
186 float key_right = KeyMap::is_event(state, InputName::PlayerRight);
187 float key_up = KeyMap::is_event(state, InputName::PlayerForward);
188 float key_down = KeyMap::is_event(state, InputName::PlayerBack);

194 // 360; log_info(" angles {} {}", xang, yang);

196 if (xang >= 135 && xang < 225) {

198 left = key_left;
199 right = key_right;
200 down = key_down;
201 up = key_up;
202 } else if (xang >= 45 && xang < 135) {
203 up = key_left;
204 down = key_right;
205 left = key_down;
206 right = key_up;
207 } else if (xang >= 315 || xang <= 45) {
208 left = key_right;
209 right = key_left;

211 down = key_up;

214 right = key_down;

216 down = key_left;

218 log_warn("reached a camera angle that has no controls enabled: {}",

221 left = 0;
222 right = 0;
223 down = 0;
224 up = 0;

228 if (right > 0) cui.write(InputName::PlayerRight);
229 if (up > 0) cui.write(InputName::PlayerForward);

238 if (rotate) cui.write(InputName::PlayerRotateFurniture);
239
240 float do_work = KeyMap::is_event(state, InputName::PlayerDoWork);
241 if (do_work > 0) cui.write(InputName::PlayerDoWork);

243 // Actually save the inputs if there were any
244 cui.publish(dt);

247inline void process_player_movement_input(std::shared_ptr<Entity> entity,
248 float dt, InputName input_name,
249 float input_amount) {
250 if (entity->is_missing<Transform>()) return;
251 Transform& transform = entity->get<Transform>();
252 std::shared_ptr<Entity> player = dynamic_pointer_cast<Entity>(entity);
253
254 if (entity->is_missing<HasBaseSpeed>()) return;
255 HasBaseSpeed& hasBaseSpeed = entity->get<HasBaseSpeed>();
256
257 const float speed = hasBaseSpeed.speed() * dt;
258 auto new_position = transform.pos();
259
260 if (input_name == InputName::PlayerLeft) {
261 new_position.x -= input_amount * speed;
262 } else if (input_name == InputName::PlayerRight) {
263 new_position.x += input_amount * speed;
264 } else if (input_name == InputName::PlayerForward) {
265 new_position.z -= input_amount * speed;
266 } else if (input_name == InputName::PlayerBack) {
267 new_position.z += input_amount * speed;
268 }
269
270 person_update_given_new_pos(entity->id, transform, player, dt, new_position,
271 new_position);
272 transform.trunc(2);

593 case InputName::PlayerPickup:
594 // grab_or_drop(entity);

607 inround::work_furniture(entity, frame_dt);

614 const float frame_dt = std::get<1>(input);

618 auto input_name = magic_enum::enum_value<InputName>(i);
619 bool was_pressed = input_set.test(i);
620 if (was_pressed) {
621 _proc_single_input_name(entity, input_name, frame_dt);

623 i++;
