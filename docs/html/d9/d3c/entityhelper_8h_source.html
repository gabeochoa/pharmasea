24// return true if the item has collision and is currently collidable
25[[nodiscard]] inline bool is_collidable(std::shared_ptr<Entity> entity) {
26 // by default we disable collisions when you are holding something
27 // since its generally inside your bounding box
28 if (entity->has<CanBeHeld>()) {
29 return !entity->get<CanBeHeld>().is_held();
30 }
31
32 if (entity->has<IsSolid>()) {
33 return true;
34 }
35
36 // if you are a ghost player and are currently not a ghost
37 // then you are collidable
38 if (entity->has<CanBeGhostPlayer>()) {
39 return entity->get<CanBeGhostPlayer>().is_not_ghost();
40 }
41 return false;
42}

154 template<typename T>
155 static std::shared_ptr<T> getMatchingEntityInFront(
156 vec2 pos, //
157 float range, //
158 Transform::FrontFaceDirection direction, //
159 std::function<bool(std::shared_ptr<T>)> filter //
160 ) {
161 TRACY_ZONE_SCOPED;
162 M_ASSERT(range > 0,
163 fmt::format("range has to be positive but was {}", range));
164
165 int cur_step = 0;
166 while (cur_step <= range) {
167 auto tile =
168 Entity::tile_infront_given_pos(pos, cur_step, direction);
169
170 for (auto& e : get_entities()) {
171 auto current_entity = dynamic_pointer_cast<T>(e);
172 if (!current_entity) continue;
173 if (!filter(current_entity)) continue;
174
175 float cur_dist = vec::distance(
176 current_entity->template get<Transform>().as2(), tile);
177 // outside reach
178 if (abs(cur_dist) > 1) continue;
179 // this is behind us
180 if (cur_dist < 0) continue;
181
182 if (vec::to2(current_entity->template get<Transform>()
183 .snap_position()) == vec::snap(tile)) {
184 return current_entity;
185 }
186 }
187 cur_step++;
188 }
189 return {};
190 }

192 template<typename T>
193 static constexpr std::shared_ptr<T> getClosestMatchingEntity(
194 vec2 pos, float range, std::function<bool(std::shared_ptr<T>)> filter) {
195 float best_distance = range;
196 std::shared_ptr<T> best_so_far;
197 for (auto& e : get_entities()) {
198 auto s = dynamic_pointer_cast<T>(e);
199 if (!s) continue;
200 if (!filter(s)) continue;
201 float d = vec::distance(pos, e->get<Transform>().as2());
202 if (d > range) continue;
203 if (d < best_distance) {
204 best_so_far = s;
205 best_distance = d;
206 }
207 }
208 return best_so_far;
209 }

246 // each target get and path find runs through all entities

248 static inline bool isWalkableRawEntities(const vec2& pos) {
249 TRACY_ZONE_SCOPED;
250 auto bounds = get_bounds({pos.x, 0.f, pos.y}, {TILESIZE});
251 bool hit_impassible_entity = false;
252 forEachEntity([&](auto entity) {
253 if (is_collidable(entity) && entity->collides(bounds)) {
254 hit_impassible_entity = true;
255 return ForEachFlow::Break;
256 }
257 return ForEachFlow::Continue;
258 });
259 return !hit_impassible_entity;
