128 template<typename T>
129 static std::shared_ptr<T> getMatchingEntityInFront(
130 vec2 pos, //
131 float range, //
132 Entity::FrontFaceDirection direction, //
133 std::function<bool(std::shared_ptr<T>)> filter //
134 ) {
135 TRACY_ZONE_SCOPED;
136 M_ASSERT(range > 0,
137 fmt::format("range has to be positive but was {}", range));
138
139 int cur_step = 0;
140 while (cur_step <= range) {
141 auto tile =
142 Entity::tile_infront_given_pos(pos, cur_step, direction);
143
144 for (auto& e : get_entities()) {
145 auto current_entity = dynamic_pointer_cast<T>(e);
146 if (!current_entity) continue;
147 if (!filter(current_entity)) continue;
148
149 float cur_dist =
150 vec::distance(vec::to2(current_entity->position), tile);
151 // outside reach
152 if (abs(cur_dist) > 1) continue;
153 // this is behind us
154 if (cur_dist < 0) continue;
155
156 if (vec::to2(vec::snap(current_entity->position)) ==
157 vec::snap(tile)) {
158 return current_entity;
159 }
160 }
161 cur_step++;
162 }
163 return {};
164 }

166 template<typename T>
167 static constexpr std::shared_ptr<T> getClosestMatchingEntity(
168 vec2 pos, float range, std::function<bool(std::shared_ptr<T>)> filter) {
169 float best_distance = range;
170 std::shared_ptr<T> best_so_far;
171 for (auto& e : get_entities()) {
172 auto s = dynamic_pointer_cast<T>(e);
173 if (!s) continue;
174 if (!filter(s)) continue;
175 float d = vec::distance(pos, vec::to2(e->position));
176 if (d > range) continue;
177 if (d < best_distance) {
178 best_so_far = s;
179 best_distance = d;
180 }
181 }
182 return best_so_far;
183 }

220 // each target get and path find runs through all entities
221 // so this will just get slower and slower over time
222 static inline bool isWalkableRawEntities(const vec2& pos) {
223 TRACY_ZONE_SCOPED;
224 auto bounds = get_bounds({pos.x, 0.f, pos.y}, {TILESIZE});
225 bool hit_impassible_entity = false;
226 forEachEntity([&](auto entity) {
227 if (entity->is_collidable() && entity->collides(bounds)) {
228 hit_impassible_entity = true;
229 return ForEachFlow::Break;
230 }
231 return ForEachFlow::Continue;
232 });
233 return !hit_impassible_entity;
234 }
