26// return true if the item has collision and is currently collidable
27[[nodiscard]] inline bool is_collidable(std::shared_ptr<Entity> entity) {
28 if (!entity) return false;
29
30 // by default we disable collisions when you are holding something
31 // since its generally inside your bounding box
32 if (entity->has<CanBeHeld>()) {
33 return !entity->get<CanBeHeld>().is_held();
34 }
35
36 if (entity->has<IsSolid>()) {
37 return true;
38 }
39
40 // if you are a ghost player and are currently not a ghost
41 // then you are collidable
42 if (entity->has<CanBeGhostPlayer>()) {
43 return entity->get<CanBeGhostPlayer>().is_not_ghost();
44 }
45 return false;
46}

171 template<typename T>
172 static std::shared_ptr<T> getMatchingEntityInFront(
173 vec2 pos, //
174 float range, //
175 Transform::FrontFaceDirection direction, //
176 std::function<bool(std::shared_ptr<T>)> filter //
177 ) {
178 TRACY_ZONE_SCOPED;
179 VALIDATE(range > 0,
180 fmt::format("range has to be positive but was {}", range));
181
182 int cur_step = 0;
183 while (cur_step <= range) {
184 auto tile =
185 Transform::tile_infront_given_pos(pos, cur_step, direction);
186
187 for (auto& e : get_entities()) {
188 auto current_entity = dynamic_pointer_cast<T>(e);
189 if (!current_entity) continue;
190 if (!filter(current_entity)) continue;
191
192 // all entitites should have transforms but just in case
193 if (current_entity->template is_missing<Transform>()) {
194 log_warn("component {} is missing transform",
195 current_entity->id);
196 continue;
197 }
198
199 Transform& transform =
200 current_entity->template get<Transform>();
201
202 float cur_dist = vec::distance(transform.as2(), tile);
203 // outside reach
204 if (abs(cur_dist) > 1) continue;
205 // this is behind us
206 if (cur_dist < 0) continue;
207
208 // TODO add a snap_as2() function to transform
209 if (vec::to2(transform.snap_position()) == vec::snap(tile)) {
210 return current_entity;
211 }
212 }
213 cur_step++;
214 }
215 return {};
216 }

218 template<typename T>
219 static constexpr std::shared_ptr<T> getClosestMatchingEntity(
220 vec2 pos, float range, std::function<bool(std::shared_ptr<T>)> filter) {
221 float best_distance = range;
222 std::shared_ptr<T> best_so_far;
223 for (auto& e : get_entities()) {
224 auto s = dynamic_pointer_cast<T>(e);
225 if (!s) continue;
226 if (!filter(s)) continue;
227 float d = vec::distance(pos, e->get<Transform>().as2());
228 if (d > range) continue;
229 if (d < best_distance) {
230 best_so_far = s;
231 best_distance = d;
232 }
233 }
234 return best_so_far;
235 }

310 // each target get and path find runs through all entities
311 // so this will just get slower and slower over time
312 static inline bool isWalkableRawEntities(const vec2& pos) {
313 TRACY_ZONE_SCOPED;
314 bool hit_impassible_entity = false;
315 forEachEntity([&](auto entity) {
316 if (!is_collidable(entity)) return ForEachFlow::Continue;
317 if (vec::distance(entity->template get<Transform>().as2(), pos) <
318 TILESIZE / 2.f) {
319 hit_impassible_entity = true;
320 return ForEachFlow::Break;
321 }
322 return ForEachFlow::Continue;
323 });
324 return !hit_impassible_entity;
325 }
