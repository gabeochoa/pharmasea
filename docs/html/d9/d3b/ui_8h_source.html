292 const auto _button_pressed = [](const uuid id) {
293 // check click
294 if (has_kb_focus(id) && get().pressed(InputName::WidgetPress)) {
295 return true;
296 }

320 }

324bool button_list(const Widget& widget, const std::vector<std::string>& options,
325 int* selectedIndex, bool* hasFocus) {
326 init_widget(widget, __FUNCTION__);
327 auto state = get().widget_init<ButtonListState>(widget.id);
328 if (selectedIndex) state->selected.set(*selectedIndex);
329
330 // TODO this assert isn't working
331 // M_ASSERT(widget.growflags & GrowFlags::Column,
332 // "button lists must have growflags Column");
333
334 // TODO :HASFOCUS do we ever want to read the value
335 // or do we want to reset focus each frame
336 // if (hasFocus) state->hasFocus.set(*hasFocus);
337 state->hasFocus.set(false);
338 auto pressed = false;
339
340 get().push_parent(widget.me);
341 std::vector<std::shared_ptr<Widget>> children;
342 for (int i = 0; i < (int) options.size(); i++) {
343 auto option = options[i];
344 std::shared_ptr<Widget> button_w = get().make_temp_widget(
345 new Widget(MK_UUID_LOOP(widget.id.ownerLayer, widget.id, i),
346 {
347 .mode = Percent,
348 .value = 1.f,
349 .strictness = 1.f,
350 },
351 {
352 .mode = Percent,
353 .value = 1.f / options.size(),
354 .strictness = 1.f,
355 }));
356
357 if (button(*button_w, option)) {
358 state->selected = static_cast<int>(i);
359 pressed = true;
360 }
361 children.push_back(button_w);
362 }
363 get().pop_parent();
364
365 // NOTE: hasFocus is generally a readonly variable because
366 // we dont insert it into state on button_list start see :HASFOCUS
367 //
368 // We use it here because we know that if its true, the caller
369 // is trying to force this to have focus OR we had focus
370 // last frame.
371 //
372 bool somethingFocused = hasFocus ? *hasFocus : false;
373 // NOTE: this exists because we only to be able to move
374 // up and down if we are messing with the button list directly
375 // It would be annoying to be focused on the textfield (e.g.)
376 // and pressing up accidentally would unfocus that and bring you to some
377 // random button list somewhere**
378 //
379 // ** in this situation they have to be visible, so no worries about
380 // arrow keying your way into a dropdown
381 //
382 for (size_t i = 0; i < options.size(); i++) {
383 somethingFocused |= has_kb_focus(children[i]->id);
384 }
385
386 if (somethingFocused) {
387 if (get().pressed(InputName::ValueUp)) {
388 state->selected = state->selected - 1;
389 if (state->selected < 0) state->selected = 0;
390 get().kb_focus_id = children[state->selected]->id;
391 }
392
393 if (get().pressed(InputName::ValueDown)) {
394 state->selected = state->selected + 1;
395 if (state->selected > (int) options.size() - 1)
396 state->selected = static_cast<int>(options.size() - 1);
397 get().kb_focus_id = children[state->selected]->id;
398 }
399 }
400
401 // NOTE: This has to be after value changes so that
402 // hasFocus is handed back up to its parent correctly
403 // this allows the dropdown to know when none of its children have focus
404 // --
405 // doing this above the keypress, allows a single frame
406 // where neither the dropdown parent nor its children are in focus
407 // and that causes the dropdown to close on selection change which isnt
408 // what we want
409 for (size_t i = 0; i < options.size(); i++) {
410 // if you got the kb focus somehow
411 // (ie by clicking or tabbing)
412 // then we will just make you selected
413 if (has_kb_focus(children[i]->id))
414 state->selected = static_cast<int>(i);
415 state->hasFocus = state->hasFocus | has_kb_focus(children[i]->id);
416 }
417
418 if (state->hasFocus) get().kb_focus_id = children[state->selected]->id;
419 if (hasFocus) *hasFocus = state->hasFocus;
420 if (selectedIndex) *selectedIndex = state->selected;
421 return pressed;
422}

522bool slider(const Widget& widget, bool vertical, float* value, float mnf,
523 float mxf) {
524 const auto _slider_render = [](Widget* widget_ptr, const bool vertical,
525 const float value) {
526 Widget& widget = *widget_ptr;
527
528 _draw_focus_ring(widget);
529 get().draw_widget(widget, theme::Usage::Primary); // slider rail
530
531 // slide
532 Rectangle rect(widget.rect);
533 const float maxScale = 0.8f;
534 const float pos_offset =
535 value * (vertical ? rect.height * maxScale : rect.width * maxScale);
536
537 rect = {
538 vertical ? rect.x : rect.x + pos_offset,
539 vertical ? rect.y + pos_offset : rect.y,
540 vertical ? rect.width : rect.width / 5.f,
541 vertical ? rect.height / 5.f : rect.height,
542 };
543
544 get().draw_widget_rect(rect, theme::Usage::Accent);
545 };
546
547 const auto _slider_value_management = [](const Widget* widget,
548 bool vertical, float* value,
549 float mnf, float mxf) {
550 auto state = get().get_widget_state<SliderState>(widget->id);
551
552 bool value_changed = false;
553 if (has_kb_focus(widget->id)) {
554 if (get().is_held_down(InputName::ValueRight)) {
555 state->value = state->value + 0.005f;
556 if (state->value > mxf) state->value = mxf;
557
558 (*value) = state->value;
559 value_changed = true;
560 }
561 if (get().is_held_down(InputName::ValueLeft)) {
562 state->value = state->value - 0.005f;
563 if (state->value < mnf) state->value = mnf;
564 (*value) = state->value;
565 value_changed = true;
566 }
567 }
568
569 if (get().active_id == widget->id) {
570 get().kb_focus_id = widget->id;
571 float v;
572 if (vertical) {
573 v = (get().mouse_info.pos.y - widget->rect.y) /
574 widget->rect.height;
575 } else {
576 v = (get().mouse_info.pos.x - widget->rect.x) /
577 widget->rect.width;
578 }
579 if (v < mnf) v = mnf;
580 if (v > mxf) v = mxf;
581 if (v != *value) {
582 state->value = v;
583 (*value) = state->value;
584 value_changed = true;
585 }
586 }
587 state->value.changed_since = value_changed;
588 };
589
590 UIContext::LastFrame lf = init_widget(widget, __FUNCTION__);
591 // TODO be able to scroll this bar with the scroll wheel
592 auto state = get().widget_init<SliderState>(widget.id);
593 bool changed_previous_frame = state->value.changed_since;
594 state->value.changed_since = false;
595 if (value) state->value.set(*value);
596
597 //
598 if (!lf.rect.has_value()) return changed_previous_frame;
599
600 widget.me->rect = lf.rect.value();
601 try_to_grab_kb(widget.id);
602 active_if_mouse_inside(widget.id, widget.rect);
603 _slider_render(widget.me, vertical, state->value.asT());
604 handle_tabbing(widget.id);
605 _slider_value_management(widget.me, vertical, value, mnf, mxf);
606
607 return changed_previous_frame;
608}

614 auto _textfield_render = [](Widget* widget_ptr,

617 Widget& widget = *widget_ptr;
618
619 TextfieldValidationDecisionFlag validationFlag =

621 : TextfieldValidationDecisionFlag::None;
622
623 bool is_invalid = validation_test(
624 validationFlag, TextfieldValidationDecisionFlag::Invalid);
625 auto focus_color =
626 is_invalid ? theme::Usage::Error : theme::Usage::Accent;
627 _draw_focus_ring(widget, focus_color);
628 // Background
629 get().draw_widget(widget, is_active_and_hot(widget.id)
630 ? theme::Usage::Secondary
631 : theme::Usage::Primary);
632
633 bool shouldWriteCursor = has_kb_focus(widget.id) && state->showCursor;
634 std::string focusStr = shouldWriteCursor ? "_" : "";
635 std::string focused_content =
636 fmt::format("{}{}", state->buffer.asT(), focusStr);
637
638 auto text_color = is_invalid ? theme::Usage::Error : theme::Usage::Font;
639 get()._draw_text(widget.rect, focused_content, text_color);

642 const auto _textfield_value_management =
643 [](const Widget* widget, TextFieldValidationFn validation) {
644 auto state = get().get_widget_state<TextfieldState>(widget->id);
645
646 auto validationFlag = validation
647 ? validation(state->buffer.asT())
648 : TextfieldValidationDecisionFlag::None;
649
650 state->cursorBlinkTime = state->cursorBlinkTime + 1;
651 if (state->cursorBlinkTime > 60) {
652 state->cursorBlinkTime = 0;
653 state->showCursor = !state->showCursor;
654 }
655
656 bool changed = false;
657
658 if (has_kb_focus(widget->id)) {
659 if (get().keychar != int()) {
660 if (validation_test(
661 validationFlag,
662 TextfieldValidationDecisionFlag::StopNewInput)) {
663 } else {
664 state->buffer.asT().append(
665 std::string(1, (char) get().keychar));
666 changed = true;
667 }
668 }
669 if (get().pressed(InputName::WidgetBackspace) ||
670 get().is_held_down_debounced(InputName::WidgetBackspace)) {
671 if (state->buffer.asT().size() > 0) {
672 state->buffer.asT().pop_back();
673 }
674 changed = true;
675 }
676
677 if (get().is_held_down(InputName::WidgetCtrl)) {
678 if (get().pressed(InputName::WidgetPaste)) {
679 auto clipboard = ext::get_clipboard_text();
680
681 // make a copy so we can text validation
682 std::string post_paste(state->buffer.asT());
683 post_paste.append(clipboard);
684
685 auto vflag =
686 validation ? validation(post_paste)
687 : TextfieldValidationDecisionFlag::None;
688 // TODO we should paste the amount that fits. but we
689 // dont know what the max length actually is, so we
690 // would need to remove one letter at a time until we
691 // hit something valid.
692
693 // TODO we probably should give some kind of visual
694 // error to the user that you cant paste right now
695 bool should_append = !validation_test(
696 vflag,
697 TextfieldValidationDecisionFlag::StopNewInput);
698
699 // commit the copy-paste
700 if (should_append) state->buffer.asT() = post_paste;
701 }
702 changed = true;
703 }
704 }
705
706 if (get().mouse_info.leftDown && is_active_and_hot(widget->id)) {
707 get().kb_focus_id = widget->id;
708 }
709 state->buffer.changed_since = changed;
710 };
711
712 UIContext::LastFrame lf = init_widget(widget, __FUNCTION__);
713 auto state = get().widget_init<TextfieldState>(widget.id);
714 bool changed_previous_frame = state->buffer.changed_since;
715
716 if (!content.empty()) state->buffer = content;
717 state->buffer.changed_since = false;
718
719 if (!lf.rect.has_value()) return changed_previous_frame;
720
721 widget.me->rect = lf.rect.value();
722 try_to_grab_kb(widget.id);
723 active_if_mouse_inside(widget.id, widget.rect);
724 _textfield_render(widget.me, validation);
725 _textfield_value_management(widget.me, validation);
726 handle_tabbing(widget.id);
727
728 content = state->buffer;
729 return changed_previous_frame;
730}
