66inline bool canvisit(const vec2& pos) {
67 auto is_collidable = [](Entity* entity) {
68 if (!entity) return false;
69 return (entity->has<IsSolid>());
70 };
71 bool hit_impassible_entity = false;
72 for (auto entity : ents) {
73 if (!is_collidable(entity)) continue;
74
75 if (vec::distance(entity->template get<Transform>().as2(), pos) <
76 TILESIZE / 2.f) {
77 hit_impassible_entity = true;
78 break;
79 }
80 continue;
81 }
