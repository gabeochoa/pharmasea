103 void forEachInputInMap(std::function<void(Event&)> cb) const {
104 for (const auto& fm_kv : mapping) {
105 for (const auto& lm_kv : fm_kv.second) {
106 for (const auto& input : lm_kv.second) {
107 std::visit(
108 util::overloaded{
109 [&](int keycode) {
110 if (visit_key(keycode) > 0.f) {
111 KeyPressedEvent* event =
112 new KeyPressedEvent(keycode, 0);
113 cb(*event);
114 delete event;
115 }
116 },
117 [&](GamepadAxisWithDir axis_with_dir) {
118 float res = visit_axis(axis_with_dir);
119 if (res > 0.f) {
120 GamepadAxisMovedEvent* event =
121 new GamepadAxisMovedEvent(
122 GamepadAxisWithDir(
123 {.axis = axis_with_dir.axis,
124 .dir = res}));
125
126 cb(*event);
127 delete event;
128 }
129 },
130 [&](GamepadButton button) {
131 if (visit_button(button) > 0.f) {
132 GamepadButtonPressedEvent* event =
133 new GamepadButtonPressedEvent(button);
134 cb(*event);
135 delete event;
136 }
137 },
138 [](auto) {}},
139 input);
140 }
141 }
142 }

145 [[nodiscard]] static float is_event(const menu::State& state,
146 const InputName& name) {
147 const AnyInputs valid_inputs = KeyMap::get_valid_inputs(state, name);
148
149 float value = 0.f;
150 for (auto& input : valid_inputs) {
151 value = fmax(value,
152 std::visit(util::overloaded{
153 [](int keycode) {
154 return visit_key_down(keycode);
155 },
156 [](GamepadAxisWithDir axis_with_dir) {
157 return visit_axis(axis_with_dir);
158 },
159 [](GamepadButton button) {
160 return visit_button_down(button);
161 },
162 [](auto) {}},
163 input));
164 }
165 return value;
166 }

234 static bool does_layer_map_contain_key(const menu::State& state,
235 int keycode) {
236 // We dont even have this Layer Map
237 if (!KeyMap::get().mapping.contains(state)) return false;
238 const LayerMapping layermap = KeyMap::get().mapping[state];
239 bool contains = false;

244 util::overloaded{[&](int k) { return k == keycode; },

352 [[nodiscard]] static float visit_axis(GamepadAxisWithDir axis_with_dir) {
353 // Note: this one is a bit more complex because we have to check if you
354 // are pushing in the right direction while also checking the magnitude
355 float mvt = ext::get_gamepad_axis_movement(0, axis_with_dir.axis);
356 // Note: The 0.25 is how big the deadzone is
357 // TODO consider making the deadzone configurable?
358 if (util::sgn(mvt) == axis_with_dir.dir && abs(mvt) > 0.25f) {
359 return abs(mvt);
360 }
361 return 0.f;
362 }

364 [[nodiscard]] static float visit_button(GamepadButton button) {
365 return IsGamepadButtonPressed(0, button) ? 1.f : 0.f;
366 }

368 [[nodiscard]] static float visit_button_down(GamepadButton button) {
369 return IsGamepadButtonDown(0, button) ? 1.f : 0.f;
370 }
