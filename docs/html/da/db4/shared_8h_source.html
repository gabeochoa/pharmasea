181template<typename S>
182void serialize(S& s, ClientPacket& packet) {
183 s.value4b(packet.channel);
184 s.value4b(packet.client_id);
185 s.value4b(packet.msg_type);
186 s.ext(packet.msg,
187 bitsery::ext::StdVariant{
188 [](S& s, ClientPacket::AnnouncementInfo& info) {
189 s.text1b(info.message, MAX_ANNOUNCEMENT_LENGTH);
190 s.value4b(info.type);
191 },
192 [](S& s, ClientPacket::PlayerJoinInfo& info) {
193 s.container4b(info.all_clients, MAX_CLIENTS);
194 s.value1b(info.is_you);
195 s.value8b(info.hashed_version);
196 s.value4b(info.client_id);
197 s.text1b(info.username, MAX_NAME_LENGTH);
198 },
199 [](S& s, ClientPacket::PlayerLeaveInfo& info) {
200 s.container4b(info.all_clients, MAX_CLIENTS);
201 s.value4b(info.client_id);
202 },
203 [](S& s, ClientPacket::PlayerControlInfo& info) {
204 s.container(
205 info.inputs, MAX_INPUTS, [](S& sv, UserInput& input) {
206 sv.ext(
207 input,
208 bitsery::ext::StdTuple{
209 [](auto& s, menu::State& o) { s.value4b(o); },
210 [](auto& s, game::State& o) { s.value4b(o); },
211 [](auto& s, InputName& o) { s.value4b(o); },
212 [](auto& s, float& o) { s.value4b(o); }});
213 });
214 },
215 [](S& s, ClientPacket::GameStateInfo& info) {
216 s.value4b(info.host_menu_state);
217 s.value4b(info.host_game_state);
218 },
219 [](S& s, ClientPacket::MapInfo& info) { s.object(info.map); },
220 [](S& s, ClientPacket::PlayerInfo& info) {
221 s.text1b(info.username, MAX_NAME_LENGTH);
222 s.value4b(info.location[0]);
223 s.value4b(info.location[1]);
224 s.value4b(info.location[2]);
225 s.value4b(info.facing_direction);
226 },
227 });
