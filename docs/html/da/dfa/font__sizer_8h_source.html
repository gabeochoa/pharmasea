70 [[nodiscard]] float get_font_size_impl(const std::string& content,
71 float width, float height,
72 float spacing) const {
73 float font_size = 10.0f;
74 float last_size = 10.0f;
75 vec2 size;
76
77 // NOTE: if you are looking at a way to speed this up switch to using
78 // powers of two and theres no need to ceil or cast to int. also
79 // shifting ( font_size <<= 1) is a clean way to do this
80 //
81
82 // lets see how big can we make the size before growing
83 // outside our bounds
84 do {
85 last_size = font_size;
86 // the smaller the number we multiply by (>1) the better fitting the
87 // text will be
88 font_size = ceilf(font_size * 1.05f);
89 log_trace("measuring for {}", font_size);
90 size = MeasureTextEx(font, content.c_str(), font_size, spacing);
91 log_trace("got {},{} for {} and {},{} and last was: {}", size.x,
92 size.y, font_size, width, height, last_size);
93 } while (size.x <= width && size.y <= height);
94
95 // return the last one that passed
96 return last_size;
97 }

99 [[nodiscard]] float get_font_size(const std::string& content, float width,

103 auto hash = std::hash<FZInfo>()(fzinfo);
104 if (!_font_size_memo.contains(hash)) {
105 _font_size_memo[hash] =
106 get_font_size_impl(content, width, height, spacing);
107 } else {
108 log_trace("found value in cache");
109 }
110 float result = _font_size_memo[hash];
111 log_trace("cache value for '{}' was {}", content, result);
