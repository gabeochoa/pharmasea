99 bool run() {
100 if (!running) return false;
101
102 auto poll_incoming_messages = [&]() {
103 ISteamNetworkingMessage *incoming_msg = nullptr;
104 int num_msgs = interface->ReceiveMessagesOnConnection(
105 connection, &incoming_msg, 1);
106 if (num_msgs == 0) return;
107 if (num_msgs == -1) {
108 M_ASSERT(false, "Failed checking for messages");
109 return;
110 }
111
112 std::string cmd;
113 cmd.assign((const char *) incoming_msg->m_pData,
114 incoming_msg->m_cbSize);
115 incoming_msg->Release();
116 this->process_message_cb(cmd);
117 };
118 auto poll_connection_state_changes = [&]() {
119 callback_instance = this;
120 interface->RunCallbacks();
121 };
122
123 poll_incoming_messages();
124 poll_connection_state_changes();
125
126 return true;
127 }

185 void on_steam_network_connection_status_changed(
186 SteamNetConnectionStatusChangedCallback_t *info) {
187 log_info("client on stream network connection status changed");
188 M_ASSERT(info->m_hConn == connection ||
189 connection == k_HSteamNetConnection_Invalid,
190 "Connection Status Error");
191
192 switch (info->m_info.m_eState) {
193 case k_ESteamNetworkingConnectionState_None:
194 // NOTE: We will get callbacks here when we destroy connections.
195 // You can ignore these.
196 break;
197
198 case k_ESteamNetworkingConnectionState_ClosedByPeer:
199 case k_ESteamNetworkingConnectionState_ProblemDetectedLocally: {
200 send_leave_info_request();
201 running = false;
202 // Print an appropriate message
203 if (info->m_eOldState ==
204 k_ESteamNetworkingConnectionState_Connecting) {
205 // Note: we could distinguish between a timeout, a rejected
206 // connection, or some other transport problem.
207 log_warn(
208 "We sought the remote host, yet our "
209 "efforts were met with defeat. ({})",
210 info->m_info.m_szEndDebug);
211 } else if (
212 info->m_info.m_eState ==
213 k_ESteamNetworkingConnectionState_ProblemDetectedLocally) {
214 log_warn(
215 "Alas, troubles beset us; we have lost contact with "
216 "the host. ({})",
217 info->m_info.m_szEndDebug);
218 } else {
219 // NOTE: We could check the reason code for a normal
220 // disconnection
221 log_info("The host hath bidden us farewell. ({})",
222 info->m_info.m_szEndDebug);
223 }
224
225 // Clean up the connection. This is important!
226 // The connection is "closed" in the network sense, but
227 // it has not been destroyed. We must close it on our end, too
228 // to finish up. The reason information do not matter in this
229 // case, and we cannot linger because it's already closed on the
230 // other end, so we just pass 0's.
231 interface->CloseConnection(info->m_hConn, 0, nullptr, false);
232 connection = k_HSteamNetConnection_Invalid;
233 break;
234 }
235
236 case k_ESteamNetworkingConnectionState_Connecting:
237 // We will get this callback when we start connecting.
238 // We can ignore this.
239 break;
240
241 case k_ESteamNetworkingConnectionState_Connected:
242 log_info("Connected to server OK");
243 send_join_info_request();
244 break;
245
246 default:
247 // Silences -Wswitch
248 break;
249 }
250 }
