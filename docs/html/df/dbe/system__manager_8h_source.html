68 void on_game_state_change(game::State new_state, game::State old_state) {

153 void process_state_change(
154 const std::vector<std::shared_ptr<Entity>>& entities, float dt) {
155 if (state_transitioned_round_to_planning) {
156 state_transitioned_round_to_planning = false;
157 for (auto& entity : entities) {
158 system_manager::delete_held_items_when_leaving_inround(entity);
159 }
160 }
161
162 if (state_transitioned_planning_to_round) {
163 state_transitioned_planning_to_round = false;
164 for (auto& entity : entities) {
165 system_manager::handle_autodrop_furniture_when_exiting_planning(
166 entity);
167 }
168 }
169
170 // All transitions
171 for (auto& entity : entities) {
172 system_manager::refetch_dynamic_model_names(entity, dt);
173 }
