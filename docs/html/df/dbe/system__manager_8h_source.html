145inline void person_update_given_new_pos(int id, Transform& transform,
146 std::shared_ptr<Person> person,
147 float dt, vec3 new_pos_x,
148 vec3 new_pos_z) {
149 int facedir_x = -1;
150 int facedir_z = -1;
151
152 vec3 delta_distance_x = new_pos_x - transform.raw_position;
153 if (delta_distance_x.x > 0) {
154 facedir_x = Transform::FrontFaceDirection::RIGHT;
155 } else if (delta_distance_x.x < 0) {
156 facedir_x = Transform::FrontFaceDirection::LEFT;
157 }
158
159 vec3 delta_distance_z = new_pos_z - transform.raw_position;
160 if (delta_distance_z.z > 0) {
161 facedir_z = Transform::FrontFaceDirection::FORWARD;
162 } else if (delta_distance_z.z < 0) {
163 facedir_z = Transform::FrontFaceDirection::BACK;
164 }
165
166 if (facedir_x == -1 && facedir_z == -1) {
167 // do nothing
168 } else if (facedir_x == -1) {
169 transform.face_direction =
170 static_cast<Transform::FrontFaceDirection>(facedir_z);
171 } else if (facedir_z == -1) {
172 transform.face_direction =
173 static_cast<Transform::FrontFaceDirection>(facedir_x);
174 } else {
175 transform.face_direction =
176 static_cast<Transform::FrontFaceDirection>(facedir_x | facedir_z);
177 }
178
179 // TODO what is this for
180 // this->get<Transform>().face_direction =
181 // Transform::FrontFaceDirection::BACK &
182 // Transform::FrontFaceDirection::LEFT;
183
184 // TODO this should be a component
185 {
186 // horizontal check
187 auto new_bounds_x = get_bounds(new_pos_x, transform.size);
188 // vertical check
189 auto new_bounds_y = get_bounds(new_pos_z, transform.size);
190
191 bool would_collide_x = false;
192 bool would_collide_z = false;
193 std::weak_ptr<Entity> collided_entity_x;
194 std::weak_ptr<Entity> collided_entity_z;
195 EntityHelper::forEachEntity([&](auto entity) {
196 if (id == entity->id) {
197 return EntityHelper::ForEachFlow::Continue;
198 }
199 if (!is_collidable(entity)) {
200 return EntityHelper::ForEachFlow::Continue;
201 }
202 if (!is_collidable(person)) {
203 return EntityHelper::ForEachFlow::Continue;
204 }
205 if (CheckCollisionBoxes(new_bounds_x, entity->bounds())) {
206 would_collide_x = true;
207 collided_entity_x = entity;
208 }
209 if (CheckCollisionBoxes(new_bounds_y, entity->bounds())) {
210 would_collide_z = true;
211 collided_entity_z = entity;
212 }
213 // Note: if these are both true, then we definitely dont need to
214 // keep going and can break early, otherwise we should check the
215 // rest to make sure
216 if (would_collide_x && would_collide_z) {
217 return EntityHelper::ForEachFlow::Break;
218 }
219 return EntityHelper::ForEachFlow::None;
220 });
221
222 if (!would_collide_x) {
223 transform.raw_position.x = new_pos_x.x;
224 }
225 if (!would_collide_z) {
226 transform.raw_position.z = new_pos_z.z;
227 }
228
229 // This value determines how "far" to impart a push force on the
230 // collided entity
231 const float directional_push_modifier = 1.0f;
232
233 // Figure out if there's a more graceful way to "jitter" things
234 // around each other
235 const float tile_div_push_mod = TILESIZE / directional_push_modifier;
236
237 if (would_collide_x || would_collide_z) {
238 if (auto entity_x = collided_entity_x.lock()) {
239 // TODO remove this check since we can just put CanBePushed
240 // on the person entity and replace with a has<> check
241 if (auto person_ptr_x = dynamic_cast<Person*>(entity_x.get())) {
242 CanBePushed& cbp = entity_x->get<CanBePushed>();
243 const float random_jitter = randSign() * TILESIZE / 2.0f;
244 if (facedir_x & Transform::FrontFaceDirection::LEFT) {
245 cbp.pushed_force.x += tile_div_push_mod;
246 cbp.pushed_force.z += random_jitter;
247 }
248 if (facedir_x & Transform::FrontFaceDirection::RIGHT) {
249 cbp.pushed_force.x -= tile_div_push_mod;
250 cbp.pushed_force.z += random_jitter;
251 }
252 }
253 }
254 if (auto entity_z = collided_entity_z.lock()) {
255 // TODO remove this check since we can just put CanBePushed
256 // on the person entity and replace with a has<> check
257 if (auto person_ptr_z = dynamic_cast<Person*>(entity_z.get())) {
258 CanBePushed& cbp = entity_z->get<CanBePushed>();
259 const float random_jitter = randSign() * TILESIZE / 2.0f;
260 if (facedir_z & Transform::FrontFaceDirection::FORWARD) {
261 cbp.pushed_force.x += random_jitter;
262 cbp.pushed_force.z += tile_div_push_mod;
263 }
264 if (facedir_z & Transform::FrontFaceDirection::BACK) {
265 cbp.pushed_force.x += random_jitter;
266 cbp.pushed_force.z -= tile_div_push_mod;
267 }
268 }
269 }
270 }
271 }

274inline void collect_user_input(std::shared_ptr<Entity> entity, float dt) {
275 if (!entity->has<CollectsUserInput>()) return;
276 CollectsUserInput& cui = entity->get<CollectsUserInput>();
277
278 // Theres no players not in game menu state,
279 const menu::State state = menu::State::Game;
280
281 float left = KeyMap::is_event(state, InputName::PlayerLeft);
282 float right = KeyMap::is_event(state, InputName::PlayerRight);
283 if (left > 0)
284 cui.inputs.push_back({state, InputName::PlayerLeft, left, dt});
285 if (right > 0)
286 cui.inputs.push_back({state, InputName::PlayerRight, right, dt});
287
288 float up = KeyMap::is_event(state, InputName::PlayerForward);
289 float down = KeyMap::is_event(state, InputName::PlayerBack);
290 if (up > 0) cui.inputs.push_back({state, InputName::PlayerForward, up, dt});
291 if (down > 0)
292 cui.inputs.push_back({state, InputName::PlayerBack, down, dt});
293
294 bool pickup =
295 KeyMap::is_event_once_DO_NOT_USE(state, InputName::PlayerPickup);
296 if (pickup) {
297 cui.inputs.push_back({state, InputName::PlayerPickup, 1.f, dt});
298 }
299
300 bool rotate = KeyMap::is_event_once_DO_NOT_USE(
301 state, InputName::PlayerRotateFurniture);
302 if (rotate) {
303 cui.inputs.push_back(
304 {state, InputName::PlayerRotateFurniture, 1.f, dt});
305 }
306
307 float do_work = KeyMap::is_event(state, InputName::PlayerDoWork);
308 if (do_work > 0) {
309 cui.inputs.push_back({state, InputName::PlayerDoWork, 1.f, dt});
310 }

313inline void process_player_movement_input(std::shared_ptr<Entity> entity,
314 float dt, InputName input_name,
315 float input_amount) {
316 if (!entity->has<Transform>()) return;
317 Transform& transform = entity->get<Transform>();
318 std::shared_ptr<Player> player = dynamic_pointer_cast<Player>(entity);
319
320 if (!entity->has<HasBaseSpeed>()) return;
321 HasBaseSpeed& hasBaseSpeed = entity->get<HasBaseSpeed>();
322
323 const float speed = hasBaseSpeed.speed() * dt;
324 auto new_position = transform.position;
325
326 if (input_name == InputName::PlayerLeft) {
327 new_position.x -= input_amount * speed;
328 } else if (input_name == InputName::PlayerRight) {
329 new_position.x += input_amount * speed;
330 } else if (input_name == InputName::PlayerForward) {
331 new_position.z -= input_amount * speed;
332 } else if (input_name == InputName::PlayerBack) {
333 new_position.z += input_amount * speed;
334 }
335
336 person_update_given_new_pos(entity->id, transform, player, dt, new_position,
337 new_position);
338 transform.position = transform.raw_position;
339};
