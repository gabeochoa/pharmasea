145 /*
146 * Given a tile, distance, and direction, returns the location of the
147 * tile `distance` distance in front of the tile
148 *
149 * @param vec2, the starting location
150 * @param int, how far in front to go
151 * @param Transform::FrontFaceDirection, which direction to go
152 *
153 * @returns vec2 the location `distance` tiles ahead
154 * */
155 static vec2 tile_infront_given_pos(
156 vec2 tile, int distance,
157 Transform::Transform::FrontFaceDirection direction) {
158 if (direction & Transform::FORWARD) {
159 tile.y += distance * TILESIZE;
160 tile.y = ceil(tile.y);
161 }
162 if (direction & Transform::BACK) {
163 tile.y -= distance * TILESIZE;
164 tile.y = floor(tile.y);
165 }
166
167 if (direction & Transform::RIGHT) {
168 tile.x += distance * TILESIZE;
169 tile.x = ceil(tile.x);
170 }
171 if (direction & Transform::LEFT) {
172 tile.x -= distance * TILESIZE;
173 tile.x = floor(tile.x);
174 }
175 return tile;
176 }

178 void turn_to_face_pos(const vec2 goal) {
179 float theta_rad = atan2(as2().y - goal.y, as2().x - goal.x);
180 float theta_deg = util::rad2deg(theta_rad);
181 int turn_degrees = (180 - (int) theta_deg) % 360;
182 float facing = FrontFaceDirectionMap.at(face);
183 int to_rotate = (int) (turn_degrees - facing) + 90;
184 rotate_facing_clockwise(to_rotate);
185 }
