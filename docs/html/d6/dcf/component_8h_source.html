9void compare_and_validate_components(Entity* a, Entity* b) {
10 VALIDATE(a->componentSet == b->componentSet, "component sets should match");
11
12 int i = 0;
13 while (i < max_num_components) {
14 auto eC = a->componentArray[i];
15 auto e2C = b->componentArray[i];
16 if (eC == nullptr && e2C == nullptr) {
17 i++;
18 continue;
19 }
20
21 if (eC != nullptr && e2C != nullptr) {
22 i++;
23 continue;
24 }
25
26 if (eC == nullptr) {
27 VALIDATE(false, "a missing component that b has");
28 }
29
30 if (e2C == nullptr) {
31 VALIDATE(false, "b missing component that a has");
32 }
33 i++;
34 }
35}
