80 virtual vec3 get_position_after_input(UserInputs inpts) {
81 TRACY_ZONE_SCOPED;
82 // log_trace("get_position_after_input {}", inpts.size());
83 for (const UserInput& ui : inpts) {
84 const auto menu_state = std::get<0>(ui);
85 if (menu_state != state) continue;
86
87 const InputName input_name = std::get<1>(ui);
88 const float input_amount = std::get<2>(ui);
89 const float frame_dt = std::get<3>(ui);
90
91 Transform& transform = get<Transform>();
92
93 if (input_name == InputName::PlayerPickup && input_amount > 0.5f) {
94 grab_or_drop();
95 continue;
96 }
97
98 if (input_name == InputName::PlayerRotateFurniture &&
99 input_amount > 0.5f) {
100 rotate_furniture();
101 continue;
102 }
103
104 // TODO replace with correctly named input
105 if (input_name == InputName::PlayerDoWork && input_amount > 0.5f) {
106 work_furniture(frame_dt);
107 continue;
108 }
109
110 // Movement down here...
111
112 const float speed = this->base_speed() * frame_dt;
113 auto new_position = transform.position;
114
115 if (input_name == InputName::PlayerLeft) {
116 new_position.x -= input_amount * speed;
117 } else if (input_name == InputName::PlayerRight) {
118 new_position.x += input_amount * speed;
119 } else if (input_name == InputName::PlayerForward) {
120 new_position.z -= input_amount * speed;
121 } else if (input_name == InputName::PlayerBack) {
122 new_position.z += input_amount * speed;
123 }
124
125 const auto fd = get_face_direction(new_position, new_position);
126 int fd_x = std::get<0>(fd);
127 int fd_z = std::get<1>(fd);
128 update_facing_direction(fd_x, fd_z);
129 handle_collision(fd_x, new_position, fd_z, new_position);
130 transform.position = transform.raw_position;
131 }
132 return this->get<Transform>().position;
133 }
