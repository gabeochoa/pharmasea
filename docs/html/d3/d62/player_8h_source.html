89 virtual vec3 get_position_after_input(UserInputs inpts) {
90 TRACY_ZONE_SCOPED;
91 // log_trace("get_position_after_input {}", inpts.size());
92 for (const UserInput& ui : inpts) {
93 const auto menu_state = std::get<0>(ui);
94 if (menu_state != state) continue;
95
96 const InputName input_name = std::get<1>(ui);
97 const float input_amount = std::get<2>(ui);
98 const float frame_dt = std::get<3>(ui);
99
100 if (input_name == InputName::PlayerPickup && input_amount > 0.5f) {
101 grab_or_drop();
102 continue;
103 }
104
105 if (input_name == InputName::PlayerRotateFurniture &&
106 input_amount > 0.5f) {
107 rotate_furniture();
108 continue;
109 }
110
111 // TODO replace with correctly named input
112 if (input_name == InputName::PlayerDoWork && input_amount > 0.5f) {
113 work_furniture(frame_dt);
114 continue;
115 }
116
117 // Movement down here...
118
119 const float speed = this->base_speed() * frame_dt;
120 auto new_position = this->position;
121
122 if (input_name == InputName::PlayerLeft) {
123 new_position.x -= input_amount * speed;
124 } else if (input_name == InputName::PlayerRight) {
125 new_position.x += input_amount * speed;
126 } else if (input_name == InputName::PlayerForward) {
127 new_position.z -= input_amount * speed;
128 } else if (input_name == InputName::PlayerBack) {
129 new_position.z += input_amount * speed;
130 }
131
132 const auto fd = get_face_direction(new_position, new_position);
133 int fd_x = std::get<0>(fd);
134 int fd_z = std::get<1>(fd);
135 update_facing_direction(fd_x, fd_z);
136 handle_collision(fd_x, new_position, fd_z, new_position);
137 this->position = this->raw_position;
138 }
139 return this->position;
140 }
