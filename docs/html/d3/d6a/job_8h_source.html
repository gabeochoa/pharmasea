117 State _run_job_tick(const std::shared_ptr<Entity>& entity, float dt) {
118 before_each_job_tick(entity, dt);
119 switch (state) {
120 case Job::State::Initialize: {
121 return run_state_initialize(entity, dt);
122 }
123 case Job::State::HeadingToStart: {
124 return run_state_heading_to_start(entity, dt);
125 }
126 case Job::State::WorkingAtStart: {
127 return run_state_working_at_start(entity, dt);
128 }
129 case Job::State::HeadingToEnd: {
130 return run_state_heading_to_end(entity, dt);
131 }
132 case Job::State::WorkingAtEnd: {
133 return run_state_working_at_end(entity, dt);
134 }
135 case Job::State::Completed: {
136 // We dont run anything because completed is only used for
137 // marking cleanup
138 return state;
139 }
140 }
141 return state;
