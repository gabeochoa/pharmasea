17inline bool validate_ip(const std::string& ip) {
18 // TODO this should fail for 00000.00000.0000 but shouldnt
19 // this too 000000000000
20 //
21 // NOTE: this is const static cause its faster :)
22 const static std::regex ip_regex(
23 R"(^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$)");
24 return std::regex_match(ip, ip_regex);
