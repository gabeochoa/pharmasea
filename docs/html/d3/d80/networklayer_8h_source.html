26inline bool validate_ip(const std::string& ip) {
27 // TODO this should fail for 00000.00000.0000 but shouldnt
28 // this too 000000000000
29 //
30 // NOTE: this is const static cause its faster :)
31 const static std::regex ip_regex(
32 R"(^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$)");
33 return std::regex_match(ip, ip_regex);
34}
