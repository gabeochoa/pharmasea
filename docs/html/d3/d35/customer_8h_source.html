84 void wait_in_queue(float) {
85 auto wait_and_return = [&]() {
86 // Add the current job to the queue,
87 // and then add the waiting job
88 personal_queue.push(job);
89 this->job.reset(new Job({
90 .type = Wait,
91 .timeToComplete = 1.f,
92 .start = job->start,
93 .end = job->start,
94 }));
95 return;
96 };
97
98 switch (job->state) {
99 case Job::State::Initialize: {
100 announce("starting a new wait in queue job");
101
102 // Figure out which register to go to...
103
104 // TODO replace with finding the one with the least people in it
105 std::shared_ptr<Register> closest_target =
106 EntityHelper::getClosestMatchingEntity<Register>(
107 vec::to2(this->position), TILESIZE * 100.f,
108 [](auto&&) { return true; });
109
110 if (!closest_target) {
111 // TODO we need some kinda way to save this job,
112 // and come back to it later
113 // i think just putting a Job* unfinished in Job is probably
114 // enough
115 announce("Could not find a valid register");
116 job->state = Job::State::Initialize;
117 wait_and_return();
118 return;
119 }
120
121 job->data["register"] = closest_target.get();
122 Customer* me = this;
123 job->start = closest_target->get_next_queue_position(me);
124 job->end = closest_target->tile_infront(1);
125 job->spot_in_line = closest_target->position_in_line(me);
126 job->state = Job::State::HeadingToStart;
127 return;
128 }
129 case Job::State::HeadingToStart: {
130 bool arrived = navigate_to(job->start);
131 job->state = arrived ? Job::State::WorkingAtStart
132 : Job::State::HeadingToStart;
133 return;
134 }
135 case Job::State::WorkingAtStart: {
136 if (job->spot_in_line == 0) {
137 job->state = Job::State::HeadingToEnd;
138 return;
139 }
140
141 // Check the spot in front of us
142 Register* reg = static_cast<Register*>(job->data["register"]);
143 Customer* me = this;
144 int cur_spot_in_line = reg->position_in_line(me);
145
146 if (cur_spot_in_line == job->spot_in_line ||
147 !reg->can_move_up(me)) {
148 // We didnt move so just wait a bit before trying again
149 announce(fmt::format("im just going to wait a bit longer"));
150
151 // Add the current job to the queue,
152 // and then add the waiting job
153 job->state = Job::State::WorkingAtStart;
154 wait_and_return();
155 return;
156 }
157
158 // if our spot did change, then move forward
159 announce(fmt::format("im moving up to {}", cur_spot_in_line));
160
161 job->spot_in_line = cur_spot_in_line;
162
163 if (job->spot_in_line == 0) {
164 job->state = Job::State::HeadingToEnd;
165 return;
166 }
167
168 // otherwise walk up one spot
169 job->start = reg->tile_infront(job->spot_in_line);
170 job->state = Job::State::WorkingAtStart;
171 return;
172 }
173 case Job::State::HeadingToEnd: {
174 bool arrived = navigate_to(job->end);
175 job->state = arrived ? Job::State::WorkingAtEnd
176 : Job::State::HeadingToEnd;
177 return;
178 }
179 case Job::State::WorkingAtEnd: {
180 Register* reg = (Register*) job->data["register"];
181
182 if (reg->held_item == nullptr) {
183 announce("my rx isnt ready yet");
184 wait_and_return();
185 return;
186 }
187
188 auto bag = dynamic_pointer_cast<Bag>(reg->held_item);
189 if (!bag) {
190 announce("this isnt my rx (not a bag)");
191 wait_and_return();
192 return;
193 }
194
195 if (bag->empty()) {
196 announce("this bag is empty...");
197 wait_and_return();
198 return;
199 }
200
201 auto pill_bottle =
202 dynamic_pointer_cast<PillBottle>(bag->held_item);
203 if (!pill_bottle) {
204 announce("this bag doesnt have my pills");
205 wait_and_return();
206 return;
207 }
208
209 this->held_item = reg->held_item;
210 reg->held_item = nullptr;
211
212 announce("got it");
213 Customer* me = this;
214 reg->leave_line(me);
215 job->state = Job::State::Completed;
216 return;
217 }
218 case Job::State::Completed: {
219 job->state = Job::State::Completed;
220 return;
221 }
222 }
223 }

225 void wait(float dt) {
226 switch (job->state) {
227 case Job::State::Initialize: {
228 announce("starting a new wait job");
229 job->state = Job::State::HeadingToStart;
230 return;
231 }
232 case Job::State::HeadingToStart: {
233 job->state = Job::State::WorkingAtStart;
234 return;
235 }
236 case Job::State::WorkingAtStart: {
237 job->state = Job::State::HeadingToEnd;
238 return;
239 }
240 case Job::State::HeadingToEnd: {
241 bool arrived = navigate_to(job->end);
242 job->state = arrived ? Job::State::WorkingAtEnd
243 : Job::State::HeadingToEnd;
244 return;
245 }
246 case Job::State::WorkingAtEnd: {
247 job->timePassedInCurrentState += dt;
248 if (job->timePassedInCurrentState >= job->timeToComplete) {
249 job->state = Job::State::Completed;
250 return;
251 }
252 // announce(fmt::format("waiting a little longer: {} => {} ",
253 // job->timePassedInCurrentState,
254 // job->timeToComplete));
255 job->state = Job::State::WorkingAtEnd;
256 return;
257 }
258 case Job::State::Completed: {
259 job->state = Job::State::Completed;
260 return;
261 }
