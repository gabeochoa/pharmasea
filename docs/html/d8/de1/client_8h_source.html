115 void client_process_message_string(std::string msg) {
116 auto add_new_player = [&](int client_id, std::string username) {
117 if (remote_players.contains(client_id)) {
118 log_warn("Why are we trying to add {}", client_id);
119 return;
120 };
121
122 remote_players[client_id] = std::make_shared<RemotePlayer>();
123 auto rp = remote_players[client_id];
124 rp->client_id = client_id;
125 rp->get<HasName>().update(username);
126 // NOTE we add to the map directly because its colocated with
127 // the other entity info
128 map->remote_players_NOT_SERIALIZED.push_back(
129 remote_players[client_id]);
130 log_info("Adding a player {}", client_id);
131 };
132
133 auto remove_player = [&](int client_id) {
134 for (auto it = map->remote_players_NOT_SERIALIZED.begin();
135 it != map->remote_players_NOT_SERIALIZED.end(); it++) {
136 if ((*it)->client_id == client_id) {
137 map->remote_players_NOT_SERIALIZED.erase(it);
138 break;
139 }
140 }
141
142 auto rp_it = remote_players.find(client_id);
143 if (rp_it == remote_players.end()) {
144 log_warn(
145 "RemovePlayer:: Remote player doesnt exist but should: {}",
146 client_id);
147 return;
148 }
149
150 auto rp = rp_it->second;
151 if (!rp) {
152 log_warn(
153 "RemovePlayer:: Remote player exists but has null "
154 "shared_ptr",
155 client_id);
156 } else {
157 rp->cleanup = true;
158 }
159
160 remote_players.erase(client_id);
161 };
162
163 auto update_remote_player = [&](int client_id, std::string username,
164 float* location, int facing) {
165 if (!remote_players.contains(client_id)) {
166 log_warn("Remote player doesnt exist but should: {}",
167 client_id);
168 add_new_player(client_id, username);
169 }
170 auto rp = remote_players[client_id];
171 if (!rp) return;
172 rp->update_remotely(location, username, facing);
173 };
174
175 ClientPacket packet = client_p->deserialize_to_packet(msg);
176
177 switch (packet.msg_type) {
178 case ClientPacket::MsgType::Announcement: {
179 ClientPacket::AnnouncementInfo info =
180 std::get<ClientPacket::AnnouncementInfo>(packet.msg);
181 announcements.push_back(info);
182 } break;
183 case ClientPacket::MsgType::PlayerLeave: {
184 ClientPacket::PlayerLeaveInfo info =
185 std::get<ClientPacket::PlayerLeaveInfo>(packet.msg);
186 remove_player(info.client_id);
187 } break;
188 case ClientPacket::MsgType::PlayerJoin: {
189 ClientPacket::PlayerJoinInfo info =
190 std::get<ClientPacket::PlayerJoinInfo>(packet.msg);
191
192 if (info.is_you) {
193 // We are the person that joined,
194 id = info.client_id;
195 log_info("my id is {}", id);
196 add_new_player(id, client_p->username);
197 GLOBALS.set("active_camera_target",
198 remote_players[id].get());
199 }
200
201 for (auto client_id : info.all_clients) {
202 if (info.is_you && client_id == info.client_id) continue;
203 // otherwise someone just joined and we have to deal
204 // with them
205 add_new_player(client_id, info.username);
206 }
207
208 } break;
209 case ClientPacket::MsgType::GameState: {
210 ClientPacket::GameStateInfo info =
211 std::get<ClientPacket::GameStateInfo>(packet.msg);
212 // TODO do we need to clear?
213 // Menu::get().clear_history();
214 MenuState::get().set(info.host_menu_state);
215 GameState::get().set(info.host_game_state);
216 } break;
217 case ClientPacket::MsgType::PlayerLocation: {
218 ClientPacket::PlayerInfo info =
219 std::get<ClientPacket::PlayerInfo>(packet.msg);
220 update_remote_player(packet.client_id, info.username,
221 info.location, info.facing_direction);
222 } break;
223 case ClientPacket::MsgType::Map: {
224 ClientPacket::MapInfo info =
225 std::get<ClientPacket::MapInfo>(packet.msg);
226 client_entities_DO_NOT_USE = info.map.entities();
227 client_items_DO_NOT_USE = info.map.items();
228 } break;
229
230 default:
231 log_warn("Client: {} not handled yet: {} ", packet.msg_type,
232 msg);
233 break;
234 }
235 }
