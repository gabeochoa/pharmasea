50inline float compute_size_for_parent_expectation(Widget* widget,
51 int exp_index) {
52 if (widget->absolute &&
53 widget->size_expected[exp_index].mode == SizeMode::Percent) {
54 widget->print_tree();
55 VALIDATE(false, "Absolute widgets should not use Percent");
56 }
57
58 float no_change = widget->computed_size[exp_index];
59 if (!widget->parent || widget->absolute) return no_change;
60 float parent_size = widget->parent->computed_size[exp_index];
61 SizeExpectation exp = widget->size_expected[exp_index];
62 float new_size = exp.value * parent_size;
63 switch (exp.mode) {
64 case SizeMode::Percent:
65 return parent_size == -1 ? no_change : new_size;
66 default:
67 return no_change;
68 }
69}

116inline float compute_size_for_child_expectation(Widget* widget, int exp_index) {
117 log_trace("csfce {} {}", *widget, exp_index);
118 float no_change = widget->computed_size[exp_index];
119 if (widget->children.empty()) return no_change;
120
121 float total_child_size =
122 _sum_children_axis_for_child_exp(widget, exp_index);
123 float max_child_size = _max_child_size(widget, exp_index);
124
125 float expectation = total_child_size;
126
127 if (widget->growflags & GrowFlags::Column && exp_index == 0) {
128 expectation = max_child_size;
129 }
130
131 SizeExpectation exp = widget->size_expected[exp_index];
132 switch (exp.mode) {
133 case SizeMode::Children:
134 return expectation;
135 default:
136 return no_change;
137 }

201inline void _solve_error_with_optional(Widget* widget, int exp_index,
202 float* error) {
203 int num_optional_children = 0;
204 for (Widget* child : widget->children) {
205 // Dont worry about absolute positioned children
206 if (child->absolute) continue;
207
208 SizeExpectation exp = child->size_expected[exp_index];
209 if (exp.strictness == 0.f) {
210 num_optional_children++;
211 }
212 }
213 if (num_optional_children != 0) {
214 float approx_epc = *error / num_optional_children;
215 for (Widget* child : widget->children) {
216 // Dont worry about absolute positioned children
217 if (child->absolute) continue;
218
219 SizeExpectation exp = child->size_expected[exp_index];
220 if (exp.strictness == 0.f) {
221 float cur_size = child->computed_size[exp_index];
222 child->computed_size[exp_index] =
223 fmaxf(0, cur_size - approx_epc);
224 if (cur_size > approx_epc) *error -= approx_epc;
225 }
226 }
227 }

230inline void fix_violating_children(Widget* widget, int exp_index, float error,
231 int num_children) {
232 VALIDATE(num_children != 0, "Should never have zero children");
233
234 int num_strict_children = 0;
235 int num_ignorable_children = 0;
236 for (Widget* child : widget->children) {
237 SizeExpectation exp = child->size_expected[exp_index];
238 if (exp.strictness == 1.f) {
239 num_strict_children++;
240 }
241 if (child->absolute) {
242 num_ignorable_children++;
243 }
244 }
245
246 // If there is any error left,
247 // we have to take away from the allowed children;
248
249 int num_resizeable_children =
250 num_children - num_strict_children - num_ignorable_children;
251
252 // TODO we cannot enforce the assert below in the case of wrapping
253 // because the positioning happens after error correction
254 if (error > ACCEPTABLE_ERROR && num_resizeable_children == 0 &&
255 //
256 !((widget->growflags & GrowFlags::Column) ||
257 (widget->growflags & GrowFlags::Row))
258 //
259 ) {
260 widget->print_tree();
261 log_warn("Error was {}", error);
262 VALIDATE(
263 num_resizeable_children > 0,
264 "Cannot fit all children inside parent and unable to resize any of "
265 "the children");
266 }
267
268 float approx_epc = error / (1.f * std::max(1, num_resizeable_children));
269 for (Widget* child : widget->children) {
270 SizeExpectation exp = child->size_expected[exp_index];
271 if (exp.strictness == 1.f || child->absolute) {
272 continue;
273 }
274 float portion_of_error = (1.f - exp.strictness) * approx_epc;
275 float cur_size = child->computed_size[exp_index];
276 child->computed_size[exp_index] = fmaxf(0, cur_size - portion_of_error);
277 // Reduce strictness every round
278 // so that eventually we will get there
279 exp.strictness = fmaxf(0.f, exp.strictness - 0.05f);
280 child->size_expected[exp_index] = exp;
281 }

332inline void solve_violations(Widget* widget) {
333 int num_children = _get_total_number_children(widget);
334 if (num_children == 0) return;
335
336 // me -> left -> right
337
338 float my_size_x = widget->computed_size[0];
339 float my_size_y = widget->computed_size[1];
340
341 float all_children_x = _get_total_child_size_for_violations(widget, 0);
342 float error_x = all_children_x - my_size_x;
343 int i_x = 0;
344 log_trace("preopt errorx {} ", error_x);
345 while (error_x > ACCEPTABLE_ERROR) {
346 _solve_error_with_optional(widget, 0, &error_x);
347 i_x++;
348 log_trace("error x {}", error_x);
349 fix_violating_children(widget, 0, error_x, num_children);
350 all_children_x = _get_total_child_size_for_violations(widget, 0);
351 error_x = all_children_x - my_size_x;
352 if (i_x > 100) {
353 log_warn("Hit X-axis iteration limit trying to solve violations");
354 // VALIDATE(false, "hit x iteration limit trying to solve
355 // violations");
356 break;
357 }
358 }
359
360 if (error_x < 0) {
361 tax_refund(widget, 0, error_x);
362 }
363
364 float all_children_y = _get_total_child_size_for_violations(widget, 1);
365 float error_y = all_children_y - my_size_y;
366 int i_y = 0;
367 log_trace("preopt errory {} ", error_y);
368 while (error_y > ACCEPTABLE_ERROR) {
369 _solve_error_with_optional(widget, 1, &error_y);
370 i_y++;
371 log_trace("error y {}", error_y);
372 fix_violating_children(widget, 1, error_y, num_children);
373 all_children_y = _get_total_child_size_for_violations(widget, 1);
374 error_y = all_children_y - my_size_y;
375 if (i_y > 100) {
376 log_warn("Hit Y-axis iteration limit trying to solve violations");
377 // VALIDATE(false, "hit y iteration limit trying to solve
378 // violations");
379 break;
380 }
381 }
382
383 if (error_y < 0) {
384 tax_refund(widget, 1, error_y);
385 }
386
387 // Solve for children
388 for (Widget* child : widget->children) {
389 solve_violations(child);
390 }
391}

393inline void compute_relative_positions(Widget* widget) {
394 if (widget->parent == nullptr) {
395 // This already happens by default, but lets be explicit about it
396 widget->computed_relative_pos[0] = 0.f;
397 widget->computed_relative_pos[1] = 0.f;
398 }
399
400 // Assuming we dont care about things smaller than 1 pixel
401 widget->computed_size[0] = round(widget->computed_size[0]);
402 widget->computed_size[1] = round(widget->computed_size[1]);
403 float sx = widget->computed_size[0];
404 float sy = widget->computed_size[1];
405
406 float offx = 0.f;
407 float offy = 0.f;
408
409 // Represents the current wrap's largest
410 // ex. on Column mode we only care about where to start the next column
411 float col_w = 0.f;
412 float col_h = 0.f;
413
414 const auto update_max_size = [&](float cx, float cy) {
415 col_w = fmax(cx, col_w);
416 col_h = fmax(cy, col_h);
417 };
418
419 for (Widget* child : widget->children) {
420 // Dont worry about any children that are absolutely positioned
421 if (child->absolute) {
422 compute_relative_positions(child);
423 continue;
424 }
425
426 float cx = child->computed_size[0];
427 float cy = child->computed_size[1];
428
429 bool will_hit_max_x = false;
430 bool will_hit_max_y = false;
431
432 will_hit_max_x = cx + offx > sx;
433 will_hit_max_y = cy + offy > sy;
434
435 // We cant grow and are going over the limit
436 if (widget->growflags & GrowFlags::None &&
437 (will_hit_max_x || will_hit_max_y)) {
438 child->computed_relative_pos[0] = sx;
439 child->computed_relative_pos[1] = sy;
440 continue;
441 }
442
443 // We can grow vertically and current child will push us over height
444 // lets wrap
445 if (widget->growflags & GrowFlags::Column && cy + offy > sy) {
446 offy = 0;
447 offx += col_w;
448 col_w = cx;
449 }
450
451 // We can grow horizontally and current child will push us over
452 // width lets wrap
453 if (widget->growflags & GrowFlags::Row && cx + offx > sx) {
454 offx = 0;
455 offy += col_h;
456 col_h = cy;
457 }
458
459 child->computed_relative_pos[0] = offx;
460 child->computed_relative_pos[1] = offy;
461
462 // Setup for next child placement
463 if (widget->growflags & GrowFlags::Column) {
464 offy += cy;
465 }
466 if (widget->growflags & GrowFlags::Row) {
467 offx += cx;
468 }
469
470 update_max_size(cx, cy);
471 compute_relative_positions(child);
472 }

475inline void compute_rect_bounds(Widget* widget) {
476 log_trace("computing rect bounds for {}", *widget);
477 vec2 offset = vec2{0.f, 0.f};
478 Widget* parent = widget->parent;
479 if (parent) {
480 offset = offset + vec2{parent->rect.x, parent->rect.y};
481 // parent = parent->parent;
482 }
483
484 Rectangle rect;
485 rect.x = offset.x + widget->computed_relative_pos[0];
486 rect.y = offset.y + widget->computed_relative_pos[1];
487 rect.width = widget->computed_size[0];
488 rect.height = widget->computed_size[1];
489
490 widget->rect = rect;
491
492 for (Widget* child : widget->children) {
493 compute_rect_bounds(child);
494 }
495}
