42static std::vector<std::string> split_string(const std::string& str,
43 const std::string& delimiter) {
44 std::vector<std::string> strings;
45
46 std::string::size_type pos = 0;
47 std::string::size_type prev = 0;
48 while ((pos = str.find(delimiter, prev)) != std::string::npos) {
49 strings.push_back(str.substr(prev, pos - prev));
50 prev = pos + delimiter.size();
51 }
52
53 // To get the last substring (or only, if delimiter is not found)
54 strings.push_back(str.substr(prev));
55
56 return strings;
57}
