80 [[nodiscard]] bool has() const {
81 log_trace("checking component {} {} on entity {}",
82 components::get_type_id<T>(), type_name<T>(), id);
83 log_trace("your set is now {}", componentSet);
84 bool result = componentSet[components::get_type_id<T>()];
85 log_trace("and the result was {}", result);
86 return result;

109 template<typename T, typename... TArgs>
110 T& addComponent(TArgs&&... args) {
111 log_trace("adding component_id:{} {} to entity_id: {}",
112 components::get_type_id<T>(), type_name<T>(), id);
113
114 if (this->has<T>()) {
115 log_warn(
116 "This entity {}, {} already has this component attached id: "
117 "{}, "
118 "component {}",
119 this->get<DebugName>().name(), id, components::get_type_id<T>(),
120 type_name<T>());
121
122 VALIDATE(false, "duplicate component");
123 // Commented out on purpose because the assert is gonna kill the
124 // program anyway at some point we should stop enforcing it to avoid
125 // crashing the game when people are playing
126 //
127 // return this->get<T>();
128 }
129
130 T* component(new T(std::forward<TArgs>(args)...));
131 componentArray[components::get_type_id<T>()] = component;
132 componentSet[components::get_type_id<T>()] = true;
133
134 log_trace("your set is now {}", componentSet);
135
136 component->onAttach();
137
138 return *component;
139 }

152 template<typename T>
153 [[nodiscard]] T& get() const {
154 if (this->is_missing<DebugName>()) {
155 log_error(
156 "This entity is missing debugname which will cause issues for "
157 "if the get<> is missing");
158 }
159 if (this->is_missing<T>()) {
160 log_warn(
161 "This entity {} {} is missing id: {}, "
162 "component {}",
163 this->get<DebugName>().name(), id, components::get_type_id<T>(),
164 type_name<T>());
165 }
166 BaseComponent* comp = componentArray.at(components::get_type_id<T>());
167 return *static_cast<T*>(comp);
168 }

176 template<typename S>
177 void serialize(S& s) {
178 s.value4b(id);
179 s.ext(componentSet, bitsery::ext::StdBitset{});
180 s.value1b(cleanup);
181
182 s.ext(componentArray, StdMap{max_num_components},
183 [](S& sv, int& key, BaseComponent*(&value)) {
184 sv.value4b(key);
185 sv.ext(value, PointerOwner{PointerType::Nullable});
186 });
187 }
