160 virtual void turn_to_face_entity(Entity* target) {
161 if (!target) return;
162
163 // dot product visualizer https://www.falstad.com/dotproduct/
164
165 // the angle between two vecs is
166 // @ = arccos( (a dot b) / ( |a| * |b| ) )
167
168 // first get the headings and normalise so |x| = 1
169 const vec2 my_heading = vec::norm(this->get_heading());
170 const vec2 tar_heading = vec::norm(target->get_heading());
171 // dp = ( (a dot b )/ 1)
172 float dot_product = vec::dot2(my_heading, tar_heading);
173 // arccos(dp)
174 float theta_rad = acosf(dot_product);
175 float theta_deg = util::rad2deg(theta_rad);
176 int turn_degrees = (180 - (int) theta_deg) % 360;
177 // TODO fix this
178 (void) turn_degrees;
179 if (turn_degrees > 0 && turn_degrees <= 45) {
180 this->get<Transform>().face_direction =
181 static_cast<Transform::FrontFaceDirection>(0);
182 } else if (turn_degrees > 45 && turn_degrees <= 135) {
183 this->get<Transform>().face_direction =
184 static_cast<Transform::FrontFaceDirection>(90);
185 } else if (turn_degrees > 135 && turn_degrees <= 225) {
186 this->get<Transform>().face_direction =
187 static_cast<Transform::FrontFaceDirection>(180);
188 } else if (turn_degrees > 225) {
189 this->get<Transform>().face_direction =
190 static_cast<Transform::FrontFaceDirection>(270);
191 }

261 /*
262 * Given a tile, distance, and direction, returns the location of the
263 * tile `distance` distance in front of the tile
264 *
265 * @param vec2, the starting location
266 * @param int, how far in front to go
267 * @param Transform::FrontFaceDirection, which direction to go
268 *
269 * @returns vec2 the location `distance` tiles ahead
270 * */
271 static vec2 tile_infront_given_pos(
272 vec2 tile, int distance,
273 Transform::Transform::FrontFaceDirection direction) {
274 if (direction & Transform::FORWARD) {
275 tile.y += distance * TILESIZE;
276 tile.y = ceil(tile.y);
277 }
278 if (direction & Transform::BACK) {
279 tile.y -= distance * TILESIZE;
280 tile.y = floor(tile.y);
281 }
282
283 if (direction & Transform::RIGHT) {
284 tile.x += distance * TILESIZE;
285 tile.x = ceil(tile.x);
286 }
287 if (direction & Transform::LEFT) {
288 tile.x -= distance * TILESIZE;
289 tile.x = floor(tile.x);
290 }
291 return tile;
292 }
