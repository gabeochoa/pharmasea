53 template<typename T, typename... TArgs>
54 T& addComponent(TArgs&&... args) {
55 M_ASSERT(!this->has<T>(),
56 "This entity already has this component attached");
57
58 log_trace("adding component {} {} to entity {}",
59 components::get_type_id<T>(), type_name<T>(), id);
60
61 std::shared_ptr<T> component;
62 component.reset(new T(std::forward<TArgs>(args)...));
63 // TODO figure out why this causes double free
64 // component->entity = std::shared_ptr<Entity>(this);
65 componentArray[components::get_type_id<T>()] = component;
66 componentSet[components::get_type_id<T>()] = true;
67
68 log_trace("your set is now {}", componentSet);
69
70 return *component;
71 }

203 virtual void turn_to_face_entity(Entity* target) {
204 if (!target) return;
205
206 // dot product visualizer https://www.falstad.com/dotproduct/
207
208 // the angle between two vecs is
209 // @ = arccos( (a dot b) / ( |a| * |b| ) )
210
211 // first get the headings and normalise so |x| = 1
212 const vec2 my_heading = vec::norm(this->get_heading());
213 const vec2 tar_heading = vec::norm(target->get_heading());
214 // dp = ( (a dot b )/ 1)
215 float dot_product = vec::dot2(my_heading, tar_heading);
216 // arccos(dp)
217 float theta_rad = acosf(dot_product);
218 float theta_deg = util::rad2deg(theta_rad);
219 int turn_degrees = (180 - (int) theta_deg) % 360;
220 // TODO fix this
221 (void) turn_degrees;
222 if (turn_degrees > 0 && turn_degrees <= 45) {
223 this->get<Transform>().face_direction =
224 static_cast<Transform::FrontFaceDirection>(0);
225 } else if (turn_degrees > 45 && turn_degrees <= 135) {
226 this->get<Transform>().face_direction =
227 static_cast<Transform::FrontFaceDirection>(90);
228 } else if (turn_degrees > 135 && turn_degrees <= 225) {
229 this->get<Transform>().face_direction =
230 static_cast<Transform::FrontFaceDirection>(180);
231 } else if (turn_degrees > 225) {
232 this->get<Transform>().face_direction =
233 static_cast<Transform::FrontFaceDirection>(270);
234 }
235 }

323 /*
324 * Given a tile, distance, and direction, returns the location of the
325 * tile `distance` distance in front of the tile
326 *
327 * @param vec2, the starting location
328 * @param int, how far in front to go
329 * @param Transform::FrontFaceDirection, which direction to go
330 *
331 * @returns vec2 the location `distance` tiles ahead
332 * */
333 static vec2 tile_infront_given_pos(
334 vec2 tile, int distance,
335 Transform::Transform::FrontFaceDirection direction) {
336 if (direction & Transform::FORWARD) {
337 tile.y += distance * TILESIZE;
338 tile.y = ceil(tile.y);
339 }
340 if (direction & Transform::BACK) {
341 tile.y -= distance * TILESIZE;
342 tile.y = floor(tile.y);
343 }
344
345 if (direction & Transform::RIGHT) {
346 tile.x += distance * TILESIZE;
347 tile.x = ceil(tile.x);
348 }
349 if (direction & Transform::LEFT) {
350 tile.x -= distance * TILESIZE;
351 tile.x = floor(tile.x);
352 }
353 return tile;
354 }
