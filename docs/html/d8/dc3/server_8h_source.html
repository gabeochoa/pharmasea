85 void run() {
86 TRACY_ZONE_SCOPED;
87 thread_id = std::this_thread::get_id();
88 GLOBALS.set("server_thread_id", &thread_id);
89
90 using namespace std::chrono_literals;
91 auto start = std::chrono::high_resolution_clock::now();
92 auto end = start;
93 float duration = 0.f;
94 while (running) {
95 tick(duration);
96
97 do {
98 TRACY_ZONE_NAMED(server_run_wait_loop,
99 "waiting for run loop continue", true);
100 end = std::chrono::high_resolution_clock::now();
101 duration =
102 std::chrono::duration_cast<std::chrono::milliseconds>(end -
103 start)
104 .count();
105 std::this_thread::sleep_for(1ms);
106 } while (duration < 4);
107
108 start = end;
109 }
110 }

124 void tick(float dt) {
125 TRACY_ZONE_SCOPED;
126 server_p->run();
127
128 // Check to see if we have any new packets to process
129 while (!incoming_message_queue.empty()) {
130 TRACY_ZONE_NAMED(tracy_server_process, "packets to process", true);
131 log_trace("Incoming Messages {}", incoming_message_queue.size());
132 server_process_message_string(incoming_message_queue.front());
133 incoming_message_queue.pop_front();
134 }
135
136 // TODO move one of these while loops, probably the one belowVVV to a
137 // different thread
138 // This will allow us to empty the steam queue faster, and give us more
139 // control over which messages to drop
140
141 // Check to see if we have any packets to send off
142 while (!packet_queue.empty()) {
143 TRACY_ZONE_NAMED(tracy_server_fwd, "packets to fwd", true);
144 log_trace("Packets to FWD {}", packet_queue.size());
145 ClientPacket& p = packet_queue.front();
146
147 switch (p.msg_type) {
148 case ClientPacket::MsgType::GameState: {
149 ClientPacket::GameStateInfo info =
150 std::get<ClientPacket::GameStateInfo>(p.msg);
151 // TODO probably dont need this
152 current_menu_state = info.host_menu_state;
153 current_game_state = info.host_game_state;
154 } break;
155 default:
156 break;
157 }
158
159 //
160 send_client_packet_to_all(p);
161 packet_queue.pop_front();
162 }
163
164 if (MenuState::s_is_game(current_menu_state)) {
165 TRACY_ZONE(tracy_server_gametick);
166 if (next_update_timer.test(dt)) {
167 for (auto p : players) {
168 p.second->update(next_update_timer / 1000.f);
169 }
170
171 pharmacy_map->onUpdate(next_update_timer / 1000.f);
172 }
173 }
174
175 next_map_tick -= dt;
176 if (next_map_tick <= 0) {
177 send_map_state();
178 next_map_tick = next_map_tick_reset;
179 }
180 TRACY_FRAME_MARK("server::tick");
181 }

362 void server_process_message_string(const ClientMessage& client_message) {
363 TRACY_ZONE_SCOPED;
364 // Note: not using structured binding since they cannot be captured by
365 // lambda expr yet
366 const internal::Client_t& incoming_client = client_message.first;
367 const std::string& msg = client_message.second;
368
369 const ClientPacket packet = network::deserialize_to_packet(msg);
370
371 switch (packet.msg_type) {
372 case ClientPacket::MsgType::Announcement: {
373 return process_announcement_packet(incoming_client, packet);
374 } break;
375 case ClientPacket::MsgType::PlayerControl: {
376 return process_player_control_packet(incoming_client, packet);
377 } break;
378 case ClientPacket::MsgType::PlayerJoin: {
379 return process_player_join_packet(incoming_client, packet);
380 } break;
381 case ClientPacket::MsgType::PlayerLeave: {
382 return process_player_leave_packet(incoming_client, packet);
383 } break;
384 default:
385 // No clue so lets just send it to everyone except the guy that
386 // sent it to us
387 send_client_packet_to_all(
388 packet, [&](internal::Client_t& client) {
389 return client.client_id == incoming_client.client_id;
390 });
391 log_warn("Server: {} not handled yet ", packet.msg_type);
392 break;
393 }
394 }
