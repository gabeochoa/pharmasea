100 void run() {
101 TRACY_ZONE_SCOPED;
102 thread_id = std::this_thread::get_id();
103 GLOBALS.set("server_thread_id", &thread_id);
104
105 using namespace std::chrono_literals;
106 auto start = std::chrono::high_resolution_clock::now();
107 auto end = start;
108 float duration = 0.f;
109 while (running) {
110 tick(duration);
111
112 do {
113 TRACY_ZONE_NAMED(server_run_wait_loop,
114 "waiting for run loop continue", true);
115 end = std::chrono::high_resolution_clock::now();
116 duration =
117 (float)
118 std::chrono::duration_cast<std::chrono::milliseconds>(
119 end - start)
120 .count();
121 std::this_thread::sleep_for(1ms);
122 } while (duration < 4);
123
124 start = end;
125 }
126 }

132 // TODO verify that these numbers make sense, i have a feeling
133 // its not 2fps but 1/50 seconds which woudl be 0.5fps

143 void tick(float dt) {
144 TRACY_ZONE_SCOPED;
145 server_p->run();
146
147 // Check to see if we have any new packets to process
148 while (!incoming_message_queue.empty()) {
149 TRACY_ZONE_NAMED(tracy_server_process, "packets to process", true);
150 log_trace("Incoming Messages {}", incoming_message_queue.size());
151 server_process_message_string(incoming_message_queue.front());
152 incoming_message_queue.pop_front();
153 }
154
155 // TODO move one of these while loops, probably the one belowVVV to a
156 // different thread
157 // This will allow us to empty the steam queue faster, and give us more
158 // control over which messages to drop
159
160 // Check to see if we have any packets to send off
161 while (!packet_queue.empty()) {
162 TRACY_ZONE_NAMED(tracy_server_fwd, "packets to fwd", true);
163 log_trace("Packets to FWD {}", packet_queue.size());
164 ClientPacket& p = packet_queue.front();
165
166 switch (p.msg_type) {
167 case ClientPacket::MsgType::GameState: {
168 ClientPacket::GameStateInfo info =
169 std::get<ClientPacket::GameStateInfo>(p.msg);
170 // TODO probably dont need this
171 current_menu_state = info.host_menu_state;
172 current_game_state = info.host_game_state;
173 } break;
174 default:
175 break;
176 }
177
178 //
179 send_client_packet_to_all(p);
180 packet_queue.pop_front();
181 }
182
183 // TODO right now we have the run update on all the server players
184 // this kinda makes sense but most of the game doesnt need this.
185 //
186 // the big ones are held item/furniture position updates
187 // without this the position doesnt change until you drop it
188 //
189 // TODO idk the perf implications of this (map->vec)
190 std::vector<std::shared_ptr<Entity>> temp_players;
191 for (auto it = players.begin(); it != players.end(); ++it) {
192 temp_players.push_back(it->second);
193 }
194
195 if (MenuState::s_is_game(current_menu_state)) {
196 TRACY_ZONE(tracy_server_gametick);
197 if (next_update_timer.test(dt)) {
198 pharmacy_map->_onUpdate(temp_players,
199 next_update_timer / 1000.f);
200 }
201 }
202
203 next_map_tick -= dt;
204 if (next_map_tick <= 0) {
205 send_map_state();
206 next_map_tick = next_map_tick_reset;
207 }
208
209 next_player_rare_tick -= dt;
210 if (next_player_rare_tick <= 0) {
211 // TODO decide if this needs to be more / less often
212 send_player_rare_data();
213 next_player_rare_tick = next_player_rare_tick_reset;
214 }
215 TRACY_FRAME_MARK("server::tick");
216 }

417 void server_process_message_string(const ClientMessage& client_message) {
418 TRACY_ZONE_SCOPED;
419 // Note: not using structured binding since they cannot be captured by
420 // lambda expr yet
421 const internal::Client_t& incoming_client = client_message.first;
422 const std::string& msg = client_message.second;
423
424 const ClientPacket packet = network::deserialize_to_packet(msg);
425
426 // log_info("Server: recieved packet {}", packet.msg_type);
427
428 switch (packet.msg_type) {
429 case ClientPacket::MsgType::Announcement: {
430 return process_announcement_packet(incoming_client, packet);
431 } break;
432 case ClientPacket::MsgType::PlayerControl: {
433 return process_player_control_packet(incoming_client, packet);
434 } break;
435 case ClientPacket::MsgType::PlayerJoin: {
436 return process_player_join_packet(incoming_client, packet);
437 } break;
438 case ClientPacket::MsgType::PlayerLeave: {
439 return process_player_leave_packet(incoming_client, packet);
440 } break;
441 case ClientPacket::MsgType::Ping: {
442 return process_ping_message(incoming_client, packet);
443 } break;
444 // case ClientPacket::MsgType::PlayerRare: {
445 // return process_player_rare_packet(incoming_client, packet);
446 // } break;
447 default:
448 // No clue so lets just send it to everyone except the guy that
449 // sent it to us
450 send_client_packet_to_all(
451 packet, [&](internal::Client_t& client) {
452 return client.client_id == incoming_client.client_id;
453 });
454 log_warn("Server: {} not handled yet ", packet.msg_type);
455 break;
456 }
457 }
