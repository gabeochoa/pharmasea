164 std::stringstream buffer;
165 buffer << ifs.rdbuf();
166 auto buf_str = buffer.str();
167
168 bitsery::quickDeserialization<settings::InputAdapter>(
169 {buf_str.begin(), buf_str.size()}, data);
170
171 refresh_settings();
172
173 log_info("Settings Loaded: {}", data);
174 log_trace("End loading settings file");
175 ifs.close();
176 return true;
177 }
