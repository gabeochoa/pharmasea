13struct Samples {
14 std::string filename = "unknown";
15 float samples[SAMPLE_SIZE];
16 int index = 0;
17 int num_items = 0;
18 float sum = 0.0;
19
20 void add_sample(float s) {
21 index = (index + 1) % SAMPLE_SIZE;
22 num_items++;
23 sum += s;
24
25 if (num_items < SAMPLE_SIZE) {
26 samples[index] = s;
27 } else {
28 float& old = samples[index];
29 sum -= old;
30 old = s;
31 }
32 }
33
34 [[nodiscard]] float average() const {
35 return sum / std::min(num_items, SAMPLE_SIZE);
36 }
37};

84[[nodiscard]] inline const std::string computeMethodName(
85 const std::string& function, const std::string& prettyFunction,
86 const char* extra) {
87 // If the input is a constructor, it gets the beginning of
88 // the class name, not of the method. That's why later on we
89 // have to search for the first parenthesys
90 size_t locFunName = prettyFunction.find(function);
91 size_t begin = prettyFunction.rfind(" ", locFunName) + 1;
92 // Adding function.length() make
93 // this faster and also allows to
94 // handle operator parenthesis!
95 size_t end = prettyFunction.find("(", locFunName + function.length());
96 auto suffix = prettyFunction[end + 1] == ')' ? "()" : "(...)";
97 return fmt::format("{}{}{}", prettyFunction.substr(begin, end - begin),
98 suffix, extra);
99}
